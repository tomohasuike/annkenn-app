/**
* 最終更新日時だけを軽く取得する関数（フロントエンドから定期的に呼ばれる）
*/
function getLastUpdateTime() {
 try {
   const syncSs = SpreadsheetApp.openById(SYNC_SS_ID);
   const appDataSheet = syncSs.getSheetByName('AppData');
   if (appDataSheet) {
     const time = appDataSheet.getRange("D1").getValue();
     return time ? new Date(time).getTime() : 0;
   }
   return 0;
 } catch (e) {
   return 0;
 }
}

/**
* スプレッドシートから最新データと保存済みの同期データを取得
*/
function getInitialData() {
 const data = {
   projects: [],
   resources: [],
   syncData: { assignments: {}, comments: {}, customResources: {}, lastUpdate: 0 },
   userEmail: '',
   isAdmin: false
 };

 const findColumnIndex = (headers, keyword) => {
   let idx = headers.findIndex(h => h === keyword);
   if (idx !== -1) return idx;
   idx = headers.findIndex(h => h.includes(keyword));
   return idx;
 };

 try {
   // --- 1. 案件データの取得 ---
   const pSs = SpreadsheetApp.openById(PROJECT_SS_ID);
   const pSheet = pSs.getSheets().find(s => s.getName().includes('工事案件マスター'));
   if (pSheet) {
     const pValues = pSheet.getDataRange().getValues();
     const pHeaders = pValues[0].map(h => String(h).trim());
    
     const idxId = findColumnIndex(pHeaders, '管理ID');
     const idxNo = findColumnIndex(pHeaders, '工事番号');
     const idxName = findColumnIndex(pHeaders, '工事名称');
     const idxClient = findColumnIndex(pHeaders, '発注者');
     const idxSite = findColumnIndex(pHeaders, '現場名');
     const idxType = findColumnIndex(pHeaders, '区分');
     const idxStatus = findColumnIndex(pHeaders, '工事状態フラグ');
     const idxManager = findColumnIndex(pHeaders, '発注先担当者');

     const statusCol = idxStatus !== -1 ? idxStatus : pHeaders.indexOf('状態');

     for (let i = 1; i < pValues.length; i++) {
       const row = pValues[i];
       const status = statusCol !== -1 ? String(row[statusCol]) : "";
      
       if (status === '着工中' || status === '着工前') {
         data.projects.push({
           id: idxId !== -1 ? String(row[idxId]) : 'p' + i,
           no: idxNo !== -1 ? String(row[idxNo]) : '',
           name: idxName !== -1 ? String(row[idxName]) : '名称未設定',
           client: idxClient !== -1 ? String(row[idxClient]) : '',
           site: idxSite !== -1 ? String(row[idxSite]) : '',
           category: idxType !== -1 ? String(row[idxType]) : '一般',
           status: status,
           manager: idxManager !== -1 ? String(row[idxManager]) : '未定'
         });
       }
     }
   }

   // --- 2. 人員データの取得 ---
   const rSs = SpreadsheetApp.openById(RESOURCE_SS_ID);
   const workerSheet = rSs.getSheets().find(s => s.getName().includes('作業員名簿'));
   if (workerSheet) {
     const wValues = workerSheet.getDataRange().getValues();
     for (let i = 1; i < wValues.length; i++) {
       const name = wValues[i][0];
       if (name && String(name).trim() !== "" && name !== "氏名" && name !== "名前") {
         data.resources.push({
           id: 'worker_' + i,
           type: 'person',
           name: String(name).trim()
         });
       }
     }
   }

   // --- 3. 車両・建機データの取得 ---
   const vehicleSheet = rSs.getSheets().find(s => s.getName().includes('車両名簿'));
   if (vehicleSheet) {
     const vValues = vehicleSheet.getDataRange().getValues();
     const vHeaders = vValues[0].map(h => String(h).trim());
    
     let idxVModel = findColumnIndex(vHeaders, '車種');
     if (idxVModel === -1) idxVModel = findColumnIndex(vHeaders, '名称');
     if (idxVModel === -1) idxVModel = 3;

     let idxVNo = findColumnIndex(vHeaders, '管理番号');
     if (idxVNo === -1) idxVNo = 1;

     for (let i = 1; i < vValues.length; i++) {
       const typeValue = String(vValues[i][1] || "").trim();
       const modelName = String(vValues[i][idxVModel] || "").trim();
       const manageNo = String(vValues[i][idxVNo] || "").trim();
      
       if (!modelName) continue;

       let type = 'vehicle';
       const isMachine = typeValue.includes('建機') || typeValue.includes('重機') ||
                         typeValue.includes('機械') || typeValue.includes('ユンボ') ||
                         typeValue.includes('フォーク') || typeValue.includes('バックホウ');
       if (isMachine) type = 'machine';
      
       data.resources.push({
         id: 'v_res_' + i,
         type: type,
         name: modelName + (manageNo ? '-' + manageNo : '')
       });
     }
   }

   // --- 4. アプリ共有データの取得 ---
   const syncSs = SpreadsheetApp.openById(SYNC_SS_ID);
   let appDataSheet = syncSs.getSheetByName('AppData') || syncSs.getSheets()[0];
  
   if (appDataSheet) {
     if (appDataSheet.getName() !== 'AppData') {
       appDataSheet.setName('AppData');
     }
    
     const savedData = appDataSheet.getRange("A1:D1").getValues()[0];
     if (savedData[0]) data.syncData.assignments = JSON.parse(savedData[0]);
     if (savedData[1]) data.syncData.comments = JSON.parse(savedData[1]);
     if (savedData[2]) data.syncData.customResources = JSON.parse(savedData[2]);
     if (savedData[3]) data.syncData.lastUpdate = new Date(savedData[3]).getTime();
   }

   return data;
 } catch (e) {
   console.error(e.message);
   throw new Error('スプレッドシートの読み込みに失敗しました: ' + e.message);
 }
}

/**
* 画面側から送られてきた配置状況などを専用スプレッドシートに保存
*/
function saveSyncData(syncDataStr) {
 try {
   const syncSs = SpreadsheetApp.openById(SYNC_SS_ID);
   let appDataSheet = syncSs.getSheetByName('AppData');
  
   if (!appDataSheet) {
     appDataSheet = syncSs.insertSheet('AppData');
   }
  
   const syncData = JSON.parse(syncDataStr);
   const now = new Date();
  
   appDataSheet.getRange("A1:D1").setValues([[
     JSON.stringify(syncData.assignments || {}),
     JSON.stringify(syncData.comments || {}),
     JSON.stringify(syncData.customResources || {}),
     now
   ]]);
  
   return { success: true, lastUpdate: now.getTime() };
 } catch(e) {
   throw new Error('データの保存に失敗しました: ' + e.message);
 }
}

      const [isSomeoneElseUpdated, setIsSomeoneElseUpdated] = useState(false);

      const fetchData = () => {
        setLoading(true);
        setError(null); 
        
        if (typeof google !== 'undefined' && google.script && google.script.run) {
          google.script.run
            .withSuccessHandler((data) => {
              try {
                if (!data) throw new Error("データの取得に失敗しました。");
                
                setProjects(data.projects || []);
                setResources(data.resources || []);

                if (data.syncData) {
                  setAssignments(data.syncData.assignments || {});
                  setComments(data.syncData.comments || {});
                  if (data.syncData.customResources) {
                    const cr = data.syncData.customResources;
                    if (Array.isArray(cr)) {
                      setCustomResources(cr);
                    } else {
                      setCustomResources(cr.list || []);
                      setTodos(cr.todos || []);
                      setMemos(cr.memos || "");
                      if (cr.categoryOrder) setCategoryOrder(cr.categoryOrder);
                      if (cr.collapsedCategories) setCollapsedCategories(cr.collapsedCategories);
                      if (cr.plannedCounts) setPlannedCounts(cr.plannedCounts);
                    }
                  }
                  // ★ サーバーの最新時間を記録
                  setServerLastUpdate(data.syncData.lastUpdate || 0);
                }
                setIsInitialLoaded(true);
                setHasUnsavedChanges(false); 
                setIsSomeoneElseUpdated(false); // 警告も消す
              } catch (e) {
                setError("データ処理エラー: " + e.message);
              }
              setLoading(false);
            })
            .withFailureHandler((err) => {
              setError(err.message || "スプレッドシートの読み込みに失敗しました。"); 
              setLoading(false);
            })
            .getInitialData();
        } else {
          setLoading(false);
          setError("実行環境が見つかりません。");
        }
      };

      useEffect(() => { fetchData(); }, []);

      const isSavingRef = useRef(false);
      const saveTimeoutRef = useRef(null);

      const saveData = useCallback(() => {
        if (!isInitialLoaded || error || isSavingRef.current || !hasUnsavedChanges) return;
        
        isSavingRef.current = true;
        setIsSaving(true);
        const syncDataStr = JSON.stringify({ 
          assignments: assignments, 
          comments: comments, 
          customResources: { 
            list: customResources, 
            todos: todos, 
            memos: memos, 
            categoryOrder: categoryOrder, 
            collapsedCategories: collapsedCategories,
            plannedCounts: plannedCounts
          }
        });
        
        if (typeof google !== 'undefined' && google.script && google.script.run) {
          google.script.run
            .withSuccessHandler((res) => { 
              setIsSaving(false); 
              setLastSaved(new Date()); 
              isSavingRef.current = false;
              setHasUnsavedChanges(false); 
              // ★ 保存成功時にサーバーの時間を更新
              if (res && res.lastUpdate) {
                setServerLastUpdate(res.lastUpdate);
              }
            })
            .withFailureHandler((err) => { 
              console.error("保存エラー:", err); 
              setIsSaving(false); 
              isSavingRef.current = false;
            })
            .saveSyncData(syncDataStr);
        } else { 
          setIsSaving(false); 
          isSavingRef.current = false;
        }
      }, [assignments, comments, customResources, todos, memos, categoryOrder, collapsedCategories, plannedCounts, isInitialLoaded, error, hasUnsavedChanges]);

      // 変更があったら1.5秒後に保存
      useEffect(() => {
        if (!isInitialLoaded) return;
        if (hasUnsavedChanges) {
           if (saveTimeoutRef.current) clearTimeout(saveTimeoutRef.current);
           saveTimeoutRef.current = setTimeout(saveData, 1500);
        }
        return () => {
            if (saveTimeoutRef.current) clearTimeout(saveTimeoutRef.current);
        };
      }, [hasUnsavedChanges, saveData, isInitialLoaded]);

      // ★ 15秒ごとのポーリング（他人の更新チェック）
      useEffect(() => {
        if (!isInitialLoaded) return;
        
        const interval = setInterval(() => {
          if (typeof google !== 'undefined' && google.script && google.script.run) {
            google.script.run.withSuccessHandler((remoteTime) => {
              if (remoteTime > serverLastUpdate) {
                // サーバーの時間が新しくなっている！
                if (hasUnsavedChanges || draggedItems || draggedProject) {
                  // 自分が編集中・ドラッグ中なら警告だけ出す
                  setIsSomeoneElseUpdated(true);
                } else {
                  // 閲覧中なら自動リロード
                  fetchData();
                }
              }
            }).getLastUpdateTime();
          }
        }, 15000); // 15秒に1回
        
        return () => clearInterval(interval);
      }, [isInitialLoaded, serverLastUpdate, hasUnsavedChanges, draggedItems, draggedProject]);


      useEffect(() => {
        localStorage.setItem('kawakitadx_collapsed_resources', JSON.stringify(collapsedResources));
      }, [collapsedResources]);

      const markAsChanged = () => setHasUnsavedChanges(true);

      const handleToggleAdmin = () => {
        if (isAdmin) {
          if (window.confirm("閲覧モードに戻ります。編集機能がロックされます。")) {
            setIsAdmin(false);
            localStorage.setItem('kawakitadx_is_admin', 'false');
          }
        } else {
          const pass = window.prompt("管理用パスワードを入力してください");
          if (pass === ADMIN_PASSWORD) {
            setIsAdmin(true);
            localStorage.setItem('kawakitadx_is_admin', 'true');
            alert("管理モードに切り替えました。編集が可能です。");
          } else if (pass !== null) {
            alert("パスワードが違います。");
          }
        }
      };

      const allResources = useMemo(() => {
        const combined = [...rawResources, ...customResources];
        return combined.map(r => {
          if (r.type === 'person' && r.name && r.name.includes('蓮池') && (r.name.includes('智雄') || r.name.includes('仁'))) {
            return { ...r, type: 'president' };
          }
          return r;
        }).filter(r => {
          if (r.type === 'vehicle' && r.name && /(ハイエース|NV|サクシード|ラッシュ)/i.test(r.name)) return false;
          return true;
        });
      }, [rawResources, customResources]);

      const sortedGroupedProjects = useMemo(() => {
        const groups = {};
        (projects || []).forEach(p => {
          if (!p) return;
          const cat = p.category || '未分類';
          const pName = p.name ? p.name.toLowerCase() : "";
          const pSite = p.site ? p.site.toLowerCase() : "";
          const pNo = p.no ? p.no.toLowerCase() : "";
          const sTerm = projectSearchTerm ? projectSearchTerm.toLowerCase() : "";
          
          const statusMatch = filterStatus === 'all' || p.status === filterStatus;
          const searchMatch = !projectSearchTerm || pName.includes(sTerm) || pSite.includes(sTerm) || pNo.includes(sTerm);
          
          if (statusMatch && searchMatch) {
            if (!groups[cat]) groups[cat] = [];
            groups[cat].push(p);
          }
        });
        const categories = Object.keys(groups);
        const finalOrder = categoryOrder.length > 0 
          ? [...categoryOrder.filter(c => categories.includes(c)), ...categories.filter(c => !categoryOrder.includes(c))]
          : categories;
        const sortedGroups = {};
        finalOrder.forEach(cat => { if (groups[cat]) sortedGroups[cat] = groups[cat]; });
        return sortedGroups;
      }, [projects, categoryOrder, filterStatus, projectSearchTerm]);

      const dates = useMemo(() => {
        const list = [];
        const start = new Date(currentDate);
        const day = start.getDay();
        const diff = start.getDate() - day + (day === 0 ? -6 : 1);
        start.setDate(diff);
        for (let i = 0; i < 7; i++) {
          list.push(new Date(start));
          start.setDate(start.getDate() + 1);
        }
        return list;
      }, [currentDate]);

      const formatDate = (date) => `${date.getFullYear()}-${date.getMonth() + 1}-${date.getDate()}`;
      const moveDate = (amount) => {
        const next = new Date(currentDate);
        next.setDate(next.getDate() + (amount * 7));
        setCurrentDate(next);
      };

      const getProjectSubInfo = (p) => {
        if (!p) return "";
        if (p.category === '川北' || p.category === 'BPe') return p.site || '現場名未設定';
        return p.client || '発注者未設定';
      };

      const handleCommentClick = (dateKey) => {
        const currentComment = comments[dateKey] || "";
        const newComment = window.prompt("この日のメモを入力してください:", currentComment);
        if (newComment !== null) {
          setComments(prev => {
            const next = { ...prev };
            if (newComment.trim() === "") delete next[dateKey];
            else next[dateKey] = newComment.trim();
            return next;
          });
          markAsChanged(); 
        }
      };

      const handlePlannedCountClick = (e, dateKey) => {
        if (e) e.stopPropagation();
        const currentCount = plannedCounts[dateKey] || "";
        setModalConfig({ isOpen: true, type: 'plannedCount', dateKey, res: null, value: currentCount });
      };

      const handleChangePartnerCount = (e, res, dateKey) => {
        e.stopPropagation();
        if (!isAdmin || isMobile) return; 
        setModalConfig({ isOpen: true, type: 'partnerCount', dateKey, res, value: res.count || 1 });
      };

      const closeCountModal = () => {
        setModalConfig(prev => ({ ...prev, isOpen: false }));
      };

      const saveCountModal = () => {
        const { type, dateKey, res, value } = modalConfig;
        const num = parseInt(value, 10);
        
        if (type === 'plannedCount') {
          setPlannedCounts(prev => {
            const next = { ...prev };
            if (isNaN(num) || num <= 0) {
              delete next[dateKey];
            } else {
              next[dateKey] = num;
            }
            return next;
          });
        } else if (type === 'partnerCount') {
          if (!isNaN(num) && num > 0) {
            setAssignments(prev => {
              const next = {...prev};
              const currentList = next[dateKey] || [];
              next[dateKey] = currentList.map(r => r.instanceId === res.instanceId ? {...r, count: num} : r);
              return next;
            });
          }
        }
        markAsChanged(); 
        closeCountModal();
      };

      const resetPlannedCount = () => {
        setPlannedCounts(prev => { 
          const next = { ...prev }; 
          delete next[modalConfig.dateKey]; 
          return next; 
        });
        markAsChanged(); 
        closeCountModal();
      };

      const handleAddCustomResource = () => {
        if (!newResName.trim()) return;
        const newRes = {
          id: `custom_${Date.now()}`,
          type: newResType,
          name: newResName.trim(),
          isCustom: true
        };
        setCustomResources(prev => [...prev, newRes]);
        setNewResName('');
        markAsChanged(); 
      };

      const handleRemoveCustomResource = (id) => {
        if (!isAdmin) return;
        setCustomResources(prev => prev.filter(r => r.id !== id));
        markAsChanged(); 
      };

      const isItemSelected = (res) => {
        const identifier = res.instanceId || res.id;
        return selectedItems.some(i => (i.instanceId || i.id) === identifier);
      };

      const toggleSelection = (e, res, sourceKey) => {
        if (!canDrag) return;
        e.stopPropagation();
        const identifier = res.instanceId || res.id;
        setSelectedItems(prev => {
          const exists = prev.find(i => (i.instanceId || i.id) === identifier);
          if (exists) return prev.filter(i => (i.instanceId || i.id) !== identifier);
          return [...prev, { ...res, sourceKey }];
        });
      };

      const handleDragStart = (e, res, sourceKey) => {
        if (!canDrag) return;
        e.stopPropagation();
        const identifier = res.instanceId || res.id;
        const alreadySelected = selectedItems.some(i => (i.instanceId || i.id) === identifier);
        if (alreadySelected) setDraggedItems(selectedItems);
        else setDraggedItems([{ ...res, sourceKey }]);
      };

      const clearSelection = () => { setSelectedItems([]); };

      const handleDropToProject = (e, p, d) => {
        if (!canDrag) return;
        e.preventDefault();
        e.stopPropagation();
        if (!draggedItems || draggedItems.length === 0) return;
        const targetDateStr = formatDate(d);
        const targetKey = `${p.id}-${targetDateStr}`;
        
        setAssignments(prev => {
          let next = { ...prev };
          draggedItems.forEach(item => {
            if (item.sourceKey && next[item.sourceKey]) {
                next[item.sourceKey] = next[item.sourceKey].filter(r => r.instanceId !== item.instanceId);
            }
            let resourceToPlace = { ...item };
            delete resourceToPlace.sourceKey;
            if (!resourceToPlace.instanceId) {
                resourceToPlace.instanceId = `${item.id}-${Date.now()}-${Math.floor(Math.random() * 100000)}`;
            }
            if (resourceToPlace.type === 'partner' && !resourceToPlace.count) {
              resourceToPlace.count = 1;
            }
            const current = next[targetKey] || [];
            next[targetKey] = [...current, resourceToPlace];
          });
          return next;
        });
        markAsChanged(); 
        setDraggedItems(null);
        setSelectedItems([]);
      };

      const handleDropToUnassigned = (e) => {
        if (!canDrag) return;
        e.preventDefault();
        e.stopPropagation();
        if (!draggedItems || draggedItems.length === 0) return;
        setAssignments(prev => {
          let next = { ...prev };
          draggedItems.forEach(item => {
            if (item.sourceKey && next[item.sourceKey]) {
                next[item.sourceKey] = next[item.sourceKey].filter(r => r.instanceId !== item.instanceId);
            }
          });
          return next;
        });
        markAsChanged(); 
        setDraggedItems(null);
        setSelectedItems([]);
      };

      const handleProjectDragStart = (e, project) => {
        if (!canDrag) return;
        e.stopPropagation();
        setDraggedProject(project);
        setDraggedItems(null); 
        clearSelection();
      };

      const handleProjectDrop = (e, targetProject) => {
        if (!canDrag) return;
        e.preventDefault();
        e.stopPropagation();
        if (!draggedProject || draggedProject.category !== targetProject.category || draggedProject.id === targetProject.id) {
          setDraggedProject(null);
          return;
        }
        setProjects(prev => {
          const next = [...prev];
          const draggedIdx = next.findIndex(p => p.id === draggedProject.id);
          const targetIdx = next.findIndex(p => p.id === targetProject.id);
          if (draggedIdx > -1 && targetIdx > -1) {
            const [removed] = next.splice(draggedIdx, 1);
            next.splice(targetIdx, 0, removed);
          }
          return next;
        });
        setDraggedProject(null);
      };

      const handleCategoryDragStart = (e, category) => { if(canDrag) setDraggedCategory(category); };
      const handleCategoryDrop = (e, targetCategory) => {
        if (!canDrag) return;
        e.preventDefault();
        if (!draggedCategory || draggedCategory === targetCategory) return;
        setCategoryOrder(prev => {
          const next = [...prev];
          const draggedIdx = next.indexOf(draggedCategory);
          const targetIdx = next.indexOf(targetCategory);
          if (draggedIdx > -1 && targetIdx > -1) {
            const [removed] = next.splice(draggedIdx, 1);
            next.splice(targetIdx, 0, removed);
          }
          return next;
        });
        markAsChanged(); 
        setDraggedCategory(null);
      };

      const toggleCategory = (category) => {
        setCollapsedCategories(prev => {
            const next = { ...prev, [category]: !prev[category] };
            markAsChanged(); 
            return next;
        });
      };

      const toggleResourceCollapse = (typeKey) => {
        setCollapsedResources(prev => ({ ...prev, [typeKey]: !prev[typeKey] }));
      };

      const addTodo = () => {
        if (!newTodo.trim() || !isAdmin) return;
        const now = new Date();
        const todo = { 
          id: Date.now(), 
          text: newTodo.trim(), 
          completed: false,
          createdAt: `${now.getMonth() + 1}/${now.getDate()} ${now.getHours()}:${String(now.getMinutes()).padStart(2, '0')}`
        };
        setTodos(prev => [...prev, todo]);
        setNewTodo("");
        markAsChanged(); 
      };

      const toggleTodo = (id) => {
        if (!isAdmin) return;
        setTodos(prev => prev.map(t => t.id === id ? { ...t, completed: !t.completed } : t));
        markAsChanged(); 
      };

      const deleteTodo = (id) => {
        if (!isAdmin) return;
        setTodos(prev => prev.filter(t => t.id !== id));
        markAsChanged(); 
      };

      const activeTodos = todos.filter(t => !t.completed);
      const completedTodos = todos.filter(t => t.completed);

      if (loading) return (
        <div className="flex flex-col items-center justify-center h-screen bg-slate-50">
          <div className="loader w-10 h-10 border-4 border-slate-200 rounded-full"></div>
          <p className="mt-4 text-slate-500 font-bold">クラウドと同期しています...</p>
        </div>
      );

      if (error) return (
        <div className="flex flex-col items-center justify-center h-screen bg-slate-50 p-6 text-center">
          <Icons.AlertCircle className="w-16 h-16 text-red-500 mb-4" />
          <h2 className="text-xl font-bold text-slate-700 mb-2">データの読み込みに失敗しました</h2>
          <p className="text-red-600 bg-red-50 p-4 rounded-lg border border-red-200 w-full max-w-lg text-sm mb-6 break-words">{error}</p>
          <button onClick={fetchData} className="bg-blue-600 text-white px-8 py-3 rounded-lg font-bold shadow-md hover:bg-blue-700 transition">再読み込み</button>
        </div>
      );

      return (
        <div className="flex flex-col bg-slate-50 text-slate-800 text-[1em] overflow-hidden" style={{ height: '100dvh' }} onClick={clearSelection}>
          {/* ★ 警告バー（他の人が更新した時） */}
          {isSomeoneElseUpdated && (
            <div className="bg-red-500 text-white text-sm font-bold p-2 text-center flex flex-wrap items-center justify-center gap-2 z-[60] shadow-md shrink-0">
              <Icons.AlertCircle className="w-4 h-4" />
              他のユーザーがデータを更新しました。上書きを防ぐため、最新のデータを再読み込みしてください。
              <button onClick={fetchData} className="ml-2 bg-white text-red-600 px-3 py-1 rounded shadow-sm hover:bg-red-50 active:bg-red-100 transition-colors">再読み込みする</button>
            </div>
          )}

          {/* ヘッダー */}
          <header className="bg-white border-b px-3 py-2 flex flex-wrap sm:flex-nowrap items-center justify-between shadow-sm shrink-0 z-50 gap-2">
            <div className="flex items-center gap-3">
              {isAdmin && (
                <button 
                  onClick={() => setShowLeftPanel(!showLeftPanel)} 
                  className={`p-1.5 rounded-md transition-colors ${showLeftPanel ? 'bg-blue-100 text-blue-600' : 'bg-slate-100 text-slate-500 hover:text-blue-600'}`}
                  title="リソース一覧の表示/非表示"
                >
                  <Icons.PanelLeft />
                </button>
              )}
              <h1 className="text-[1.15em] font-black text-slate-700 flex items-center gap-1.5 leading-none shrink-0">
                <Icons.Calendar /> 建設DX 工程管理
              </h1>
              <button onClick={fetchData} className="text-slate-400 hover:text-blue-600" title="最新データを取得"><Icons.Refresh /></button>
              <div className="text-[0.75em] bg-slate-100 px-1.5 py-0.5 rounded border border-slate-200 text-slate-500 shrink-0 hidden sm:block">
                {isSaving ? "保存中..." : (lastSaved ? `同期済 (${lastSaved.getHours()}:${String(lastSaved.getMinutes()).padStart(2, '0')})` : "待機中")}
              </div>
            </div>

            {/* スライダーコントロールパネル */}
            <div className="flex items-center gap-4 bg-slate-50 border rounded-lg px-3 py-1 shadow-inner shrink-0">
              <div className="flex items-center gap-2">
                <Icons.Type className="text-slate-400 w-3 h-3" />
                <span className="text-[0.75em] font-bold text-slate-500">文字</span>
                <input 
                  type="range" 
                  min="12" max="20" step="1" 
                  value={fontSize} 
                  onChange={(e) => setFontSize(parseInt(e.target.value))}
                  className="w-20 sm:w-24"
                  title={`文字サイズ: ${fontSize}px`}
                />
              </div>
              <div className="w-px h-4 bg-slate-300"></div>
              <div className="flex items-center gap-2">
                <Icons.Maximize2 className="text-slate-400 w-3 h-3" />
                <span className="text-[0.75em] font-bold text-slate-500">案件幅</span>
                <input 
                  type="range" 
                  min="150" max="400" step="10" 
                  value={projectColumnWidth} 
                  onChange={(e) => setProjectColumnWidth(parseInt(e.target.value))}
                  className="w-20 sm:w-24"
                  title={`案件列の幅: ${projectColumnWidth}px`}
                />
              </div>
            </div>

            <div className="flex items-center gap-2 shrink-0">
              <div 
                onClick={handleToggleAdmin}
                className={`px-3 py-1 text-[0.85em] font-bold rounded-md border cursor-pointer transition-all shadow-sm ${isAdmin ? 'bg-purple-600 text-white border-purple-700 hover:bg-purple-700' : 'bg-white text-slate-600 border-slate-300 hover:bg-slate-50'}`}
              >
                {isAdmin ? '👑 管理モード' : '👤 閲覧モード'}
              </div>

              <div className="flex border rounded-md px-1 py-0.5 bg-white items-center gap-1 shadow-sm mx-1">
                <button onClick={() => moveDate(-1)} className="p-1 hover:bg-slate-100 rounded transition-colors"><Icons.ChevronLeft /></button>
                <span className="font-bold text-[0.9em] min-w-[3.5rem] text-center">{currentDate.getMonth()+1}月</span>
                <button onClick={() => moveDate(1)} className="p-1 hover:bg-slate-100 rounded transition-colors"><Icons.ChevronRight /></button>
              </div>
              <button onClick={() => setCurrentDate(new Date())} className="bg-blue-600 text-white px-2.5 py-1 rounded-md font-bold shadow-sm text-[0.85em] hover:bg-blue-700">今日</button>
              <button 
                onClick={() => setShowRightPanel(!showRightPanel)} 
                className={`p-1.5 rounded-md transition-colors ml-1 ${showRightPanel ? 'bg-blue-100 text-blue-600' : 'bg-slate-100 text-slate-500 hover:text-blue-600'}`}
                title="管理パネルの表示/非表示"
              >
                <Icons.PanelRight />
              </button>
            </div>
          </header>

          <div className="flex flex-1 overflow-hidden relative">
            
            {/* 左サイドバー：リソース一覧（管理モードかつ表示ONの時） */}
            {isAdmin && showLeftPanel && (
              <aside className="w-56 bg-white border-r flex flex-col shadow-inner shrink-0 z-50">
                <div className="p-2 border-b bg-slate-50">
                  <h2 className="font-bold text-slate-600 mb-2 flex items-center gap-1 text-[0.85em]"><Icons.Users /> リソース</h2>
                  <input 
                    type="text" 
                    placeholder="人員・車両を検索..." 
                    className="w-full px-2 py-1 border rounded text-[0.85em] outline-none focus:ring-1 focus:ring-blue-400 mb-2" 
                    value={searchTerm} 
                    onChange={e=>setSearchTerm(e.target.value)} 
                  />
                  <div className="bg-white p-1.5 border rounded shadow-sm">
                    <div className="text-[0.7em] font-bold text-slate-400 mb-1">＋ 手入力で追加</div>
                    <input type="text" placeholder="氏名/車名..." value={newResName} onChange={e=>setNewResName(e.target.value)} className="w-full border rounded px-1.5 py-1 text-[0.8em] mb-1 outline-none"/>
                    <select value={newResType} onChange={e=>setNewResType(e.target.value)} className="w-full border rounded px-1 py-1 text-[0.8em] outline-none">
                      <option value="person">社内人員</option>
                      <option value="partner">協力会社</option>
                      <option value="vehicle">車両</option>
                      <option value="machine">建機</option>
                    </select>
                    <button onClick={handleAddCustomResource} className="w-full bg-slate-100 hover:bg-slate-200 text-slate-600 font-bold py-1 rounded text-[0.8em] mt-1 transition">追加</button>
                  </div>
                </div>
                <div className="flex-1 overflow-y-auto p-2 bg-slate-50/50">
                  {Object.entries(RESOURCE_TYPES).map(([key, info]) => {
                    const resourcesForThisKey = allResources.filter(r=>r.type===key && formatResName(r.name).includes(searchTerm));
                    if (resourcesForThisKey.length === 0) return null;
                    const isCollapsed = collapsedResources[key];
                    return (
                      <div key={key} className="mb-2">
                        <h3 
                          className="text-[0.7em] font-bold text-slate-400 mb-1 uppercase tracking-tight flex items-center gap-1 cursor-pointer hover:text-slate-600"
                          onClick={() => toggleResourceCollapse(key)}
                        >
                          <info.icon className="w-2.5 h-2.5" /> {info.label} {isCollapsed ? '▼' : '▲'}
                        </h3>
                        {!isCollapsed && (
                            <div className="space-y-0.5">
                            {resourcesForThisKey.map(res => {
                                const isSelected = selectedItems.some(i => (i.instanceId ? i.instanceId : i.id) === res.id);
                                return (
                                <div 
                                    key={res.id} 
                                    draggable={canDrag} 
                                    onClick={(e) => {
                                    if (!canDrag) return;
                                    setSelectedItems(p => {
                                        const exists = p.some(i => (i.instanceId ? i.instanceId : i.id) === res.id);
                                        if (exists) return p.filter(i => (i.instanceId ? i.instanceId : i.id) !== res.id);
                                        return [...p, {...res, sourceKey:null}];
                                    });
                                    }} 
                                    onDragStart={(e) => { 
                                    if(!canDrag) return; 
                                    e.stopPropagation(); 
                                    const exists = selectedItems.some(i => (i.instanceId ? i.instanceId : i.id) === res.id);
                                    if (exists) setDraggedItems(selectedItems);
                                    else setDraggedItems([{...res, sourceKey:null}]);
                                    }} 
                                    className={`px-1.5 py-0.5 border rounded border-l-[3px] ${info.border} font-bold text-[0.8em] flex items-center justify-between cursor-grab active:cursor-grabbing hover:bg-slate-50 transition-all ${isSelected ? 'bg-blue-50 ring-1 ring-blue-500' : 'bg-white shadow-sm'}`}
                                >
                                    <div className="flex items-center gap-1 overflow-hidden"><info.icon className={`w-2.5 h-2.5 ${info.iconColor} shrink-0`} /><span className="truncate">{formatResName(res.name)}</span></div>
                                    {res.isCustom && <button onClick={(e) => { e.stopPropagation(); handleRemoveCustomResource(res.id); }} className="text-slate-300 hover:text-red-500"><Icons.X /></button>}
                                </div>
                                );
                            })}
                            </div>
                        )}
                      </div>
                    );
                  })}
                </div>
              </aside>
            )}

            {/* 中央メインエリア */}
            <main className="flex-1 overflow-auto bg-slate-100 relative">
              <div className="flex items-center gap-4 p-1.5 bg-white border-b sticky left-0 z-30 shadow-sm w-max min-w-full">
                <select className="border rounded px-2 py-0.5 text-[0.85em] outline-none bg-slate-50 cursor-pointer text-slate-600 font-bold" value={filterStatus} onChange={(e) => setFilterStatus(e.target.value)}>
                  <option value="all">すべての状態</option><option value="着工中">着工中</option><option value="着工前">着工前</option>
                </select>
                <div className="relative">
                  <Icons.Search className="absolute left-2 top-1.5 text-slate-400 w-3 h-3" />
                  <input type="text" placeholder="案件名、現場名を検索..." className="pl-6 pr-3 py-1 border rounded text-[0.85em] outline-none focus:ring-1 focus:ring-blue-400 shadow-inner w-64" value={projectSearchTerm} onChange={(e) => setProjectSearchTerm(e.target.value)}/>
                </div>
              </div>

              <table className="w-full border-collapse bg-white table-fixed min-w-full">
                <thead className="sticky top-0 z-30 shadow-md">
                  <tr className="bg-slate-50 text-slate-500 border-b">
                    <th 
                      className="p-1.5 border-r text-left text-[0.9em] font-bold sticky left-0 z-40 bg-slate-50"
                      style={{ width: `${projectColumnWidth}px` }} 
                    >
                      工事案件名称 / 現場名
                    </th>
                    {dates.map((d, i) => {
                      const isPast = d.getTime() < today.getTime();
                      return (
                        <th key={i} className={`p-1 border-r border-b text-center min-w-[5.5rem] ${d.getTime() === today.getTime() ? 'bg-blue-600 text-white' : isPast ? 'bg-slate-200 text-slate-400' : 'bg-slate-50'} ${d.getDay() === 0 && !isPast ? 'text-red-500' : d.getDay() === 6 && !isPast ? 'text-blue-500' : ''}`}>
                          <div className="text-[1.1em] font-bold leading-tight">{d.getDate()}</div>
                          <div className="text-[0.7em] font-normal opacity-80 leading-none">{['日','月','火','水','木','金','土'][d.getDay()]}</div>
                        </th>
                      )
                    })}
                  </tr>
                  
                  {/* 未配置行 */}
                  <tr className="bg-emerald-50/80 border-b-2 border-emerald-200">
                    <th 
                      className="p-1.5 border-r sticky left-0 z-40 bg-emerald-50 text-left text-[0.8em] font-bold text-emerald-800"
                      style={{ width: `${projectColumnWidth}px` }}
                    >
                      【未配置】{ !canDrag && <span className="text-[0.625rem] font-normal text-emerald-600 ml-1">人員のみ</span> }
                    </th>
                    {dates.map((d, i) => {
                      const ds = formatDate(d);
                      const assignedIds = new Set();
                      Object.keys(assignments).forEach(k => { if (k.endsWith(`-${ds}`)) (assignments[k]||[]).forEach(r => assignedIds.add(r.id)); });
                      const unassigned = allResources.filter(r => !assignedIds.has(r.id) && r.type !== 'partner' && r.type !== 'president');
                      
                      // 閲覧モードの時は人員のみ。管理モードの時は全て表示。
                      const displayList = canDrag ? unassigned : unassigned.filter(r => r.type === 'person');
                      
                      // 過去日の判定
                      const isPast = d.getTime() < today.getTime();

                      return (
                        <td key={`unassigned-${i}`} onDragOver={e=>canDrag && e.preventDefault()} onDrop={e => canDrag && handleDropToUnassigned(e)} className={`p-0.5 border-r border-b align-top font-normal transition-colors ${isPast ? 'bg-slate-100/80 grayscale-[0.5] opacity-70' : 'bg-emerald-50/50'} ${d.getTime() === today.getTime() ? 'bg-blue-50' : ''}`}>
                          <div className="flex flex-col gap-1 min-h-[2rem]">
                              {Object.entries(RESOURCE_TYPES).map(([typeKey, typeInfo]) => {
                                  const resForType = displayList.filter(r => r.type === typeKey);
                                  if (resForType.length === 0) return null;
                                  const isCollapsed = collapsedResources[typeKey];
                                  return (
                                      <div key={`unassigned-${typeKey}`}>
                                          <div 
                                              className="text-[0.65em] font-bold text-slate-500 mb-0.5 cursor-pointer hover:text-slate-700 flex items-center"
                                              onClick={() => toggleResourceCollapse(typeKey)}
                                          >
                                              {typeInfo.label} {isCollapsed ? '▼' : '▲'}
                                          </div>
                                          {!isCollapsed && (
                                              <div className="flex flex-wrap gap-0.5">
                                                  {resForType.map(res => (
                                                      <div 
                                                          key={res.id} 
                                                          draggable={canDrag && !isPast} 
                                                          onClick={(e) => !isPast && canDrag && toggleSelection(e, res, null)} 
                                                          onDragStart={(e) => !isPast && canDrag && handleDragStart(e, res, null)} 
                                                          className={`px-1 py-0.5 rounded bg-white border border-l-[2px] ${typeInfo.border} text-slate-700 text-[0.7em] shadow-sm font-bold leading-none ${canDrag && !isPast ? 'cursor-grab' : 'cursor-default'} ${isItemSelected(res) ? 'ring-1 ring-blue-500' : ''}`}
                                                      >
                                                          <span className="truncate max-w-[5rem]">{formatResName(res.name)}</span>
                                                      </div>
                                                  ))}
                                              </div>

                                      </div>
                                  )
                              })}
                          </div>
                        </td>
                      );
                    })}
                  </tr>

                  {/* 協力会社 集計 */}
                  <tr className="bg-purple-50/50 border-b-2 border-purple-200">
                    <th 
                      className="p-1.5 border-r sticky left-0 z-40 bg-purple-50/90 text-left text-[0.8em] font-bold text-purple-800"
                      style={{ width: `${projectColumnWidth}px` }}
                    >
                      【協力会社 集計】
                    </th>
                    {dates.map((d, i) => {
                      const ds = formatDate(d);
                      const partnerTotals = {};
                      Object.keys(assignments).forEach(k => {
                        if (k.endsWith(`-${ds}`)) {
                          (assignments[k]||[]).forEach(r => {
                            if (r.type === 'partner') {
                              const name = formatResName(r.name);
                              partnerTotals[name] = (partnerTotals[name] || 0) + (r.count || 1);
                            }
                          });
                        }
                      });
                      
                      const isPast = d.getTime() < today.getTime();

                      return (
                        <th key={`partner-sum-${i}`} className={`p-0.5 border-r border-b align-top font-normal transition-colors ${isPast ? 'bg-slate-100/80 grayscale-[0.5] opacity-70' : 'bg-purple-50/20'} ${d.getTime() === today.getTime() ? 'bg-blue-50' : ''}`}>
                          <div className="flex flex-wrap gap-0.5 min-h-[1.2rem]">
                            {Object.entries(partnerTotals).map(([name, count]) => (
                              <div key={name} className="bg-purple-100 text-purple-800 text-[0.65em] font-bold px-1.5 py-0.5 rounded border border-purple-200 shadow-sm leading-none">
                                {name}:{count}名

                            ))}
                          </div>
                        </th>
                      );
                    })}
                  </tr>
                </thead>

                <tbody>
                  {/* 休暇・不在 */}
                  <tr className="bg-rose-50/40 border-b-2 border-rose-200">
                    <td 
                      className="p-1.5 border sticky left-0 z-10 bg-rose-50/90 text-left text-[0.8em] font-bold text-rose-800 shadow-[2px_0_5px_rgba(0,0,0,0.02)]"
                      style={{ width: `${projectColumnWidth}px` }}
                    >
                      【休暇・不在】
                    </td>
                    {dates.map((d, i) => {
                      const p = { id: 'vacation', name: '休暇', category: '休暇' };
                      const dk = `${p.id}-${formatDate(d)}`;
                      const al = assignments[dk] || [];
                      const cc = comments[dk]; 
                      const isPast = d.getTime() < today.getTime();

                      return (
                        <td key={i} onDragOver={e=>canDrag && e.preventDefault()} onDrop={(e) => canDrag && handleDropToProject(e, p, d)} className={`p-0.5 border align-top transition-colors relative min-h-[3rem] group/cell ${isPast ? 'bg-slate-50/80 grayscale-[0.6] opacity-70' : ''}`}>
                           <div className="absolute top-0 right-0 z-10 flex flex-col">
                            <button onClick={(e) => { e.stopPropagation(); handleCommentClick(dk); }} className={`p-0.5 transition-all ${cc ? 'text-amber-500' : 'text-slate-200 opacity-0 group-hover/cell:opacity-100 hover:text-slate-400'}`} title="メモ"><Icons.MessageSquare /></button>
                          </div>

                          <div className="flex flex-col gap-0.5 pt-1 pr-4 min-h-[2.5rem]">
                            {cc && (
                              <div className="bg-amber-100 border border-amber-200 text-amber-800 text-[0.65em] p-1 rounded leading-tight cursor-pointer shadow-sm mb-0.5 whitespace-pre-wrap" onClick={() => handleCommentClick(dk)}>{cc}</div>
                            )}

                            <div className="flex flex-wrap gap-0.5">
                              {al.map(res => (
                                <div key={res.instanceId || res.id} draggable={canDrag} onClick={(e) => canDrag && toggleSelection(e, res, dk)} onDragStart={(e) => canDrag && handleDragStart(e, res, dk)} className={`px-1 py-0.5 rounded bg-white border border-l-[2px] ${RESOURCE_TYPES[res.type]?.border || 'border-l-slate-400'} flex items-center justify-between shadow-sm text-slate-700 text-[0.7em] font-bold leading-none w-full max-w-[6.5rem] ${canDrag ? 'cursor-grab' : ''} ${isItemSelected(res) ? 'ring-1 ring-blue-500' : ''}`}>
                                  <span className="truncate">{formatResName(res.name)}</span>
                                  {isAdmin && <button onClick={(e)=>{ e.stopPropagation(); setAssignments(prev=>({...prev, [dk]: (prev[dk]||[]).filter(r=>r.instanceId!==res.instanceId)})); markAsChanged(); }} className="hover:text-red-500 text-slate-300 ml-0.5"><Icons.X /></button>}
                                </div>
                              ))}
                            </div>
                          </div>
                        </td>
                      );
                    })}
                  </tr>

                  {/* 案件グループ */}
                  {Object.entries(sortedGroupedProjects).map(([cat, list]) => {
                    const isCol = collapsedCategories[cat];
                    
                    return (
                      <React.Fragment key={cat}>
                        <tr className="bg-slate-200/60 border-b cursor-pointer hover:bg-slate-300/60 transition-colors" onClick={() => toggleCategory(cat)}>
                          <td colSpan={8} className="p-1 border font-bold text-[0.8em] text-slate-500 uppercase sticky left-0 z-10 leading-none">
                            【区分: {cat}】 {isCol && <span className="text-[0.6em] italic ml-1">※データありのみ表示中</span>}
                          </td>
                        </tr>
                        
                        {(list||[]).map(p => {
                          if (isCol) {
                            const hasData = dates.some(d => {
                              const key = `${p.id}-${formatDate(d)}`;
                              return (assignments[key] && assignments[key].length > 0) || !!plannedCounts[key] || !!comments[key];
                            });
                            if (!hasData) return null; 
                          }

                          return (
                            <tr key={p.id} className="group hover:bg-slate-50 transition border-b h-auto">
                              <td 
                                className="p-1 border sticky left-0 z-10 bg-white group-hover:bg-slate-50 shadow-[2px_0_5px_rgba(0,0,0,0.02)]" 
                                onDragOver={(e) => canDrag && e.preventDefault()} 
                                onDrop={(e) => canDrag && handleProjectDrop(e, p)}
                                style={{ width: `${projectColumnWidth}px` }} 
                              >
                                <div className="flex gap-1 items-start">
                                  {canDrag && <div className="text-slate-300 mt-0.5 cursor-grab hover:text-blue-500" draggable onDragStart={(e) => handleProjectDragStart(e, p)}><Icons.GripVertical /></div>}
                                  <div className="flex-1 overflow-hidden leading-tight">
                                    <div className="text-[0.65em] font-bold text-blue-600">[{p.no || '----'}]</div>
                                    <div className="font-bold text-[0.85em] text-slate-700 my-0.5 truncate" title={p.name}>{p.name}</div>
                                    <div className="text-[0.65em] text-slate-400 truncate max-w-full font-medium">{getProjectSubInfo(p)}</div>
                                  </div>
                                </div>
                              </td>
                              
                              {dates.map((d, i) => {
                                const dk = `${p.id}-${formatDate(d)}`;
                                const al = assignments[dk] || [];
                                const cc = comments[dk];
                                const cp = plannedCounts[dk];
                                const isPast = d.getTime() < today.getTime();
                                
                                const count = al.reduce((acc, res) => {
                                  if (res.type === 'partner') return acc + (res.count || 1);
                                  return acc + 1;
                                }, 0);

                                return (
                                  <td key={i} onDragOver={e=>canDrag && e.preventDefault()} onDrop={(e) => canDrag && handleDropToProject(e, p, d)} className={`p-0.5 border min-h-[3.5rem] align-top transition-colors relative group/cell ${isPast ? 'bg-slate-50/80 grayscale-[0.6] opacity-70' : ''} ${d.getTime() === today.getTime() ? 'bg-blue-50/40' : ''}`}>
                                    <div className="absolute top-0 right-0 z-10 flex flex-col">
                                      <button onClick={(e) => handlePlannedCountClick(e, dk)} className={`p-0.5 transition-all ${cp ? 'text-blue-600' : 'text-slate-200 opacity-0 group-hover/cell:opacity-100 hover:text-slate-400'}`} title="予定人数"><Icons.Users /></button>
                                      <button onClick={(e) => { e.stopPropagation(); handleCommentClick(dk); }} className={`p-0.5 transition-all ${cc ? 'text-amber-500' : 'text-slate-200 opacity-0 group-hover/cell:opacity-100 hover:text-slate-400'}`} title="メモ"><Icons.MessageSquare /></button>
                                    </div>
                                    
                                    <div className="flex flex-col gap-0.5 pt-0.5 pr-4">
                                      {cp && (
                                        <div className={`text-[0.65em] font-bold px-1 py-0.5 rounded flex justify-between items-center cursor-pointer shadow-sm leading-none ${count < cp ? 'bg-red-100 text-red-600 border border-red-200' : 'bg-emerald-100 text-emerald-600 border border-emerald-200'}`} onClick={(e) => handlePlannedCountClick(e, dk)}>
                                          <span>予:{cp}</span><span>({count})</span>
                                        </div>
                                      )}
                                      
                                      {cc && (
                                        <div className="bg-amber-100 border border-amber-200 text-amber-800 text-[0.65em] p-1 rounded leading-tight cursor-pointer shadow-sm mb-0.5 whitespace-pre-wrap" onClick={() => handleCommentClick(dk)}>{cc}</div>
                                      )}
                                      
                                      <div className="flex flex-wrap gap-0.5">
                                        {al.map(res => (
                                          <div key={res.instanceId || res.id} draggable={canDrag} onClick={(e) => canDrag && toggleSelection(e, res, dk)} onDragStart={(e) => canDrag && handleDragStart(e, res, dk)} className={`px-1 py-0.5 rounded bg-white border border-l-[2.5px] ${RESOURCE_TYPES[res.type]?.border || 'border-l-slate-400'} shadow-sm text-slate-700 text-[0.7em] flex items-center justify-between font-bold leading-none w-full max-w-[6.5rem] ${canDrag ? 'cursor-grab' : ''} ${isItemSelected(res) ? 'ring-1 ring-blue-500' : ''}`}>
                                            <div className="flex items-center gap-0.5 overflow-hidden">
                                              <span className="truncate">{formatResName(res.name)}</span>
                                              {res.type === 'partner' && <span className="bg-purple-100 text-purple-700 px-0.5 rounded ml-0.5 cursor-pointer" onClick={(e)=> { e.stopPropagation(); handleChangePartnerCount(e, res, dk); }}>{res.count || 1}名</span>}
                                            </div>
                                            {isAdmin && <button onClick={(e)=>{ e.stopPropagation(); setAssignments(prev=>({...prev, [dk]: (prev[dk]||[]).filter(r=>r.instanceId!==res.instanceId)})); markAsChanged(); }} className="hover:text-red-500 text-slate-300 ml-0.5"><Icons.X /></button>}
                                          </div>
                                        ))}
                                      </div>
                                    </div>
                                  </td>
                                );
                              })}
                            </tr>
                          );
                        })}
                      </React.Fragment>
                    );
                  })}
                </tbody>
              </table>
            </main>

            {/* 右サイドバー：管理パネル（表示ONの時） */}
            {showRightPanel && (
              <aside className={`inset-y-0 right-0 w-64 bg-white border-l flex flex-col shadow-inner shrink-0 text-sm z-50`}>
                <div className="p-2 border-b bg-slate-50">
                  <h2 className="font-bold text-slate-600 flex items-center gap-1.5 text-[0.85em]"><Icons.AlertCircle className="text-amber-500" /> 管理パネル</h2>
                </div>
                <div className="flex-1 overflow-y-auto p-3 space-y-4 pb-20">
                  <section>
                    <h3 className="text-[0.75em] font-bold text-slate-400 mb-1 uppercase tracking-widest flex items-center gap-1"><Icons.MessageSquare /> 共有メモ</h3>
                    <textarea 
                      value={memos} 
                      onChange={(e) => { setMemos(e.target.value); markAsChanged(); }} 
                      readOnly={!isAdmin}
                      placeholder="全社で共有するメモを入力..." 
                      className={`w-full h-32 p-2 text-[0.85em] border rounded shadow-inner outline-none focus:ring-1 focus:ring-amber-300 resize-none leading-relaxed ${!isAdmin ? 'bg-slate-50 text-slate-500' : 'bg-white text-slate-700'}`}
                    />
                  </section>

                  <section>
                    <div className="flex items-center justify-between mb-1">
                      <h3 className="text-[0.75em] font-bold text-slate-400 uppercase tracking-widest flex items-center gap-1"><Icons.CheckCircle /> TODO</h3>
                      <button onClick={() => setShowTodoHistory(!showTodoHistory)} className={`p-1 rounded transition-colors ${showTodoHistory ? 'bg-blue-100 text-blue-600' : 'bg-slate-100 text-slate-400 hover:text-slate-600'}`} title="完了履歴"><Icons.History /></button>
                    </div>
                    {!showTodoHistory ? (
                      <>
                        {isAdmin && (
                          <div className="flex gap-1 mb-2">
                            <input type="text" value={newTodo} onChange={(e) => setNewTodo(e.target.value)} onKeyPress={(e) => e.key === 'Enter' && addTodo()} placeholder="タスク追加..." className="flex-1 text-[0.85em] border rounded px-1.5 py-1 outline-none focus:ring-1 focus:ring-blue-400"/>
                            <button onClick={addTodo} className="bg-blue-600 text-white px-2 py-1 rounded shadow-sm hover:bg-blue-700"><Icons.Plus /></button>
                          </div>
                        )}
                        <div className="space-y-1">
                          {activeTodos.map(todo => (
                            <div key={todo.id} className="bg-white border border-slate-200 p-1.5 rounded flex items-start gap-1.5 shadow-sm group">
                              <input type="checkbox" checked={todo.completed} onChange={() => toggleTodo(todo.id)} disabled={!isAdmin} className="mt-0.5 cursor-pointer"/>
                              <div className="flex-1">
                                <div className="text-[0.85em] leading-tight text-slate-700">{todo.text}</div>
                                <div className="text-[0.65em] text-slate-400">{todo.createdAt}</div>
                              </div>
                              {isAdmin && <button onClick={() => deleteTodo(todo.id)} className="text-slate-300 hover:text-red-500 opacity-0 group-hover:opacity-100"><Icons.X /></button>}
                            </div>
                          ))}
                        </div>
                      </>
                    ) : (
                      <div className="space-y-1 opacity-70">
                        {completedTodos.map(todo => (
                          <div key={todo.id} className="bg-slate-50 border border-slate-100 p-1.5 rounded flex items-start gap-1.5 group">
                            <div className="flex-1">
                              <div className="text-[0.85em] leading-tight text-slate-500 line-through">{todo.text}</div>
                            </div>
                            {isAdmin && (
                              <button onClick={() => toggleTodo(todo.id)} className="text-blue-400 hover:text-blue-600"><Icons.RotateCcw /></button>
                            )}
                          </div>
                        ))}
                      </div>
                    )}
                  </section>
                </div>
              </aside>
            )}

          </div>

          <footer className="bg-white border-t px-4 py-1.5 flex justify-between items-center text-[0.75em] text-slate-400 shrink-0">
             <div className="flex gap-4">
                <span>凡例: <span className="text-red-500 font-bold">社長</span> / <span className="text-blue-500 font-bold">人員</span> / <span className="text-purple-500 font-bold">協力</span> / <span className="text-emerald-500 font-bold">車両</span></span>
                <span className="hidden sm:inline">| クラウド自動同期中</span>
             </div>
             <div className="flex items-center gap-1.5 font-bold">
               <div className={`w-1.5 h-1.5 rounded-full ${isAdmin ? 'bg-purple-500 animate-pulse' : 'bg-slate-300'}`}></div>
               <span className={isAdmin ? 'text-purple-600' : 'text-slate-500'}>{isAdmin ? '管理モード' : '閲覧モード'}</span>
             </div>
          </footer>

          {/* モーダル */}

function getDashboardData() {
 const ss = SpreadsheetApp.getActiveSpreadsheet();
  // 1. 全社員数の取得（社員シートの行数からヘッダーを引く）
 let totalEmployees = 0;
 const empSheet = ss.getSheetByName("社員");
 if (empSheet) {
   totalEmployees = Math.max(0, empSheet.getLastRow() - 1);
 }

 // 2. 回答データの集計
 let totalAnswers = 0;
 let safe = 0;
 let minor = 0;
 let major = 0;
 const list = [];
  const ansSheet = ss.getSheetByName("回答データ");
 if (ansSheet && ansSheet.getLastRow() > 1) {
   const data = ansSheet.getRange(2, 1, ansSheet.getLastRow() - 1, 8).getValues();
   totalAnswers = data.length;
  
   // 最新の回答が上に来るように後ろからループ
   for (let i = data.length - 1; i >= 0; i--) {
     const row = data[i];
     const status = row[3];
    
     if (status === '無事') safe++;
     else if (status === '軽傷') minor++;
     else if (status === '重傷') major++;
    
     // 日付のフォーマット
     let timeStr = "";
     if (row[0] instanceof Date) {
       timeStr = Utilities.formatDate(row[0], Session.getScriptTimeZone(), "yyyy/MM/dd HH:mm");
     } else {
       timeStr = row[0];
     }

     list.push({
       time: timeStr,
       name: row[1],
       email: row[2],
       status: status,
       family: row[4],
       house: row[5],
       location: row[6],
       memo: row[7]
     });
   }
 }

 // 集計結果をまとめたオブジェクトをadmin.htmlに返す
 return {
   totalEmployees: totalEmployees,
   totalAnswers: totalAnswers,
   safe: safe,
   minor: minor,
   major: major,
   list: list
 };
}

function checkEarthquakeApi() {
 const listUrl = "https://www.jma.go.jp/bosai/quake/data/list.json";
 const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('設定');
  // B8セルから対象地域を取得（例: "那須塩原市"）
 const targetRegionsStr = sheet.getRange("B8").getValue() || "";
 const targetRegions = targetRegionsStr.split(',').map(r => r.trim()).filter(r => r !== "");
  try {
   const listResponse = UrlFetchApp.fetch(listUrl);
   const listData = JSON.parse(listResponse.getContentText());
  
   if (listData && listData.length > 0) {
     const latestQuake = listData[0];
     const maxScale = latestQuake.maxI;
     const quakeId = latestQuake.eid;
     const epicentre = latestQuake.anm;
    
     const lastNotifiedId = sheet.getRange("B3").getValue();
    
     if (quakeId !== lastNotifiedId && ["6-", "6+", "7"].includes(maxScale)) {
      
       if (targetRegions.length === 0) {
          sendEmergencyChatByApi(epicentre, maxScale, "全国");
          sheet.getRange("B3").setValue(quakeId);
          return;
       }

       if (latestQuake.json) {
          const detailUrl = "https://www.jma.go.jp/bosai/quake/data/" + latestQuake.json;
          const detailResponse = UrlFetchApp.fetch(detailUrl);
          const detailData = JSON.parse(detailResponse.getContentText());
         
          let isTargetHit = false;
          let hitRegionName = "";
         
          try {
            const prefs = detailData.Body.Intensity.Observation.Pref;
           
            for (let i = 0; i < prefs.length; i++) {
              const pref = prefs[i];
             
              if (targetRegions.includes(pref.Name) && ["6-", "6+", "7"].includes(pref.MaxInt)) {
                isTargetHit = true; hitRegionName = pref.Name; break;
              }
             
              if (pref.Area) {
                for (let j = 0; j < pref.Area.length; j++) {
                  const area = pref.Area[j];
                  if (area.City) {
                    for (let k = 0; k < area.City.length; k++) {
                      const city = area.City[k];
                     
                      if ((targetRegions.includes(city.Name) || targetRegions.includes(pref.Name + city.Name))
                          && ["6-", "6+", "7"].includes(city.MaxInt)) {
                        isTargetHit = true;
                        hitRegionName = city.Name;
                        break;
                      }
                    }
                  }
                  if (isTargetHit) break;
                }
              }
              if (isTargetHit) break;
            }
          } catch(err) {
            console.error("詳細データの解析エラー: " + err);
          }
         
          if (isTargetHit) {
            sendEmergencyChatByApi(epicentre, maxScale, hitRegionName);
            sheet.getRange("B3").setValue(quakeId);
          } else {
            sheet.getRange("B3").setValue(quakeId);
          }
       }
     }
   }
 } catch (e) {
   console.error("地震情報の取得に失敗: " + e.message);
 }
}

function sendEmergencyChatByApi(epicentre, maxScale, hitRegion) {
 const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('設定');
 const webhookUrl = sheet.getRange('B1').getValue();
 const appUrl = sheet.getRange('B2').getValue();

 const scaleText = maxScale.replace("-", "弱").replace("+", "強");
  let message = "";
 if (hitRegion === "全国") {
   message = `🚨 **【緊急安否確認】**\n\n先ほど、**${epicentre}** を震源とする **最大震度${scaleText}** の地震が発生しました。\n\nご自身の安全を確保した後、直ちに以下のURLより安否状況をご報告ください。\n\n🔗 報告フォーム: ${appUrl}`;
 } else {
   message = `🚨 **【緊急安否確認】**\n\n先ほど、対象地域（**${hitRegion}**等）で **震度6弱以上** を観測する地震が発生しました。(震源: ${epicentre})\n\nご自身の安全を確保した後、直ちに以下のURLより安否状況をご報告ください。\n\n🔗 報告フォーム: ${appUrl}`;
 }

 const options = {
   "method": "post",
   "contentType": "application/json",
   "payload": JSON.stringify({ "text": message })
 };
  UrlFetchApp.fetch(webhookUrl, options);
}

function getDashboardDataStringified() {
 try {
   const ss = SpreadsheetApp.getActiveSpreadsheet();
   const timeZone = "Asia/Tokyo";
  
   // 1. 全社員リストの取得
   let allEmployees = [];
   const empSheet = ss.getSheetByName("社員");
   if (empSheet && empSheet.getLastRow() > 0) {
     const empData = empSheet.getDataRange().getValues();
     for (let i = 1; i < empData.length; i++) {
       let colA = String(empData[i][0] || "").trim();
       let colB = String(empData[i][1] || "").trim();
       if (!colA && !colB) continue;
       let email = colA.includes('@') ? colA : (colB.includes('@') ? colB : '');
       let name = colA.includes('@') ? colB : colA;
       if (email) allEmployees.push({ name: name, email: email });
     }
   }
   let totalEmployees = allEmployees.length;

   // 2. 回答データの全取得
   let allAnswers = [];
   const ansSheet = ss.getSheetByName("回答データ");
   if (ansSheet && ansSheet.getLastRow() > 1) {
     const numCols = Math.min(ansSheet.getLastColumn(), 8);
     const data = ansSheet.getRange(2, 1, ansSheet.getLastRow() - 1, numCols).getValues();
     for (let i = data.length - 1; i >= 0; i--) {
       const row = data[i];
       if (!row[0]) continue;

       let timeMs = 0;
       let timeStr = "日時不明";
       try {
         let d = row[0] instanceof Date ? row[0] : new Date(row[0]);
         timeMs = d.getTime();
         if (!isNaN(timeMs)) {
           timeStr = Utilities.formatDate(d, timeZone, "yyyy/MM/dd HH:mm");
         } else {
           timeMs = 0; timeStr = String(row[0]);
         }
       } catch (e) { timeStr = String(row[0]); }

       allAnswers.push({
         timestamp: timeMs, time: timeStr,
         name: row[1] ? String(row[1]) : "不明",
         email: row[2] ? String(row[2]) : "",
         status: row[3] ? String(row[3]) : "",
         family: row[4] ? String(row[4]) : "",
         house: row[5] ? String(row[5]) : "",
         location: row[6] ? String(row[6]) : "",
         memo: row[7] ? String(row[7]) : ""
       });
     }
   }

   // 3. 通知履歴の取得
   let events = [];
   const historySheet = ss.getSheetByName("通知履歴");
   if (historySheet && historySheet.getLastRow() > 1) {
     const historyData = historySheet.getRange(2, 1, historySheet.getLastRow() - 1, 2).getValues();
     for (let i = 0; i < historyData.length; i++) {
       const hDate = historyData[i][0];
       if (hDate) {
         let timeMs = 0;
         try {
            let d = hDate instanceof Date ? hDate : new Date(hDate);
            timeMs = d.getTime();
         } catch(e) {}
         if(!isNaN(timeMs) && timeMs > 0) {
           events.push({
             dateStr: String(hDate), // Date型を使わず文字列化
             timeMs: timeMs,
             type: historyData[i][1] ? String(historyData[i][1]) : "通知"
           });
         }
       }
     }
     events.sort((a, b) => b.timeMs - a.timeMs);
   }

   // 4. イベントごとの集計
   function createSummary(name, answers) {
     let safe = 0, minor = 0, major = 0;
     let uniqueAnswers = [];
     let seenEmails = new Set();
    
     for (let i = 0; i < answers.length; i++) {
       let ans = answers[i];
       if (ans.email && !seenEmails.has(ans.email)) {
         seenEmails.add(ans.email);
         uniqueAnswers.push(ans);
         if (ans.status === '無事') safe++;
         else if (ans.status === '軽傷') minor++;
         else if (ans.status === '重傷') major++;
       }
     }
     let unresponded = allEmployees.filter(emp => emp.email && !seenEmails.has(emp.email));
     return { name: name, totalEmployees: totalEmployees, totalAnswers: uniqueAnswers.length, safe: safe, minor: minor, major: major, list: uniqueAnswers, unresponded: unresponded };
   }

   const eventsData = [];
   eventsData.push(createSummary("すべての回答（全体）", allAnswers));

   for (let i = 0; i < events.length; i++) {
     let currentEvent = events[i];
     let nextEvent = events[i - 1];
     let eventName = Utilities.formatDate(new Date(currentEvent.timeMs), timeZone, "yyyy/MM/dd HH:mm") + " (" + currentEvent.type + ")";
     let filteredAnswers = allAnswers.filter(ans => {
       let isAfterCurrent = ans.timestamp >= currentEvent.timeMs;
       let isBeforeNext = nextEvent ? ans.timestamp < nextEvent.timeMs : true;
       return isAfterCurrent && isBeforeNext;
     });
     eventsData.push(createSummary(eventName, filteredAnswers));
   }

   // 【極めて重要】データを文字列(JSON)に圧縮して返すことでGASの通信エラーを強制突破する
   return JSON.stringify({ eventsData: eventsData });

 } catch (error) {
   return JSON.stringify({ error: error.message });
 }
}

function checkEarthquakeApi() {
 const listUrl = "https://www.jma.go.jp/bosai/quake/data/list.json";
 const ss = SpreadsheetApp.getActiveSpreadsheet();
 const sheet = ss.getSheetByName('設定');
 const targetRegionsStr = sheet.getRange("B8").getValue() || "";
 const targetRegions = targetRegionsStr.split(',').map(r => r.trim()).filter(r => r !== "");
  try {
   const listResponse = UrlFetchApp.fetch(listUrl);
   const listData = JSON.parse(listResponse.getContentText());
   if (listData && listData.length > 0) {
     const latestQuake = listData[0];
     const maxScale = latestQuake.maxI;
     const quakeId = latestQuake.eid;
     const epicentre = latestQuake.anm;
     const lastNotifiedId = sheet.getRange("B3").getValue();
    
     if (quakeId !== lastNotifiedId && ["6-", "6+", "7"].includes(maxScale)) {
       if (targetRegions.length === 0) {
          sendEmergencyChatByApi(epicentre, maxScale, "全国");
          sheet.getRange("B3").setValue(quakeId);
          return;
       }
       if (latestQuake.json) {
          const detailUrl = "https://www.jma.go.jp/bosai/quake/data/" + latestQuake.json;
          const detailResponse = UrlFetchApp.fetch(detailUrl);
          const detailData = JSON.parse(detailResponse.getContentText());
          let isTargetHit = false;
          let hitRegionName = "";
          try {
            const prefs = detailData.Body.Intensity.Observation.Pref;
            for (let i = 0; i < prefs.length; i++) {
              const pref = prefs[i];
              if (targetRegions.includes(pref.Name) && ["6-", "6+", "7"].includes(pref.MaxInt)) {
                isTargetHit = true; hitRegionName = pref.Name; break;
              }
              if (pref.Area) {
                for (let j = 0; j < pref.Area.length; j++) {
                  const area = pref.Area[j];
                  if (area.City) {
                    for (let k = 0; k < area.City.length; k++) {
                      const city = area.City[k];
                      if ((targetRegions.includes(city.Name) || targetRegions.includes(pref.Name + city.Name))
                          && ["6-", "6+", "7"].includes(city.MaxInt)) {
                        isTargetHit = true; hitRegionName = city.Name; break;
                      }
                    }
                  }
                  if (isTargetHit) break;
                }
              }
              if (isTargetHit) break;
            }
          } catch(err) {}
         
          if (isTargetHit) {
            sendEmergencyChatByApi(epicentre, maxScale, hitRegionName);
            sheet.getRange("B3").setValue(quakeId);
          } else {
            sheet.getRange("B3").setValue(quakeId);
          }
       }
     }
   }
 } catch (e) {}
}

       function initDashboard(data) {
           allEventsData = data.eventsData;

       function renderDashboard(data) {

function doGet() {
 try {
   return HtmlService.createTemplateFromFile('index').evaluate()
     .setTitle('人工集計ダッシュボード')
     .addMetaTag('viewport', 'width=device-width, initial-scale=1');
 } catch (e) {
   return HtmlService.createHtmlOutput("エラー: index.htmlが見つかりません。" + e.toString());
 }
}

function findColMulti(headers, names) {
 for (const name of names) {
   const idx = headers.findIndex(h => String(h).replace(/\s/g, "").includes(name));
   if (idx !== -1) return idx;
 }
 return -1;
}

function getProjectList() {
 const resourceSS = SpreadsheetApp.openById(SS_RESOURCES_ID);
 const masterSS = SpreadsheetApp.openById(SS_MASTER_ID);
  let nippoSht = resourceSS.getSheetByName("工事日報") || resourceSS.getSheetByName("業務予定");
 let masterSht = masterSS.getSheetByName("工事案件マスター") || masterSS.getSheetByName("案件マスタ");
  if (!nippoSht || !masterSht) return [];

 const nippoData = nippoSht.getDataRange().getValues();
 const nHead = nippoData[0];
 const colProjInNippo = findColMulti(nHead, ["工事案件", "工事案件ID", "案件ID"]);
  const existingIds = new Set();
 if (colProjInNippo !== -1) {
   for (let i = 1; i < nippoData.length; i++) {
     const pId = String(nippoData[i][colProjInNippo]);
     if (pId && pId !== "undefined") existingIds.add(pId);
   }
 }

 const mData = masterSht.getDataRange().getValues();
 const mHead = mData[0];
 const colStatus = findColMulti(mHead, ["工事状態フラグ", "状態", "ステータス"]);
 const colKubun = findColMulti(mHead, ["区分"]);
 const colNo = findColMulti(mHead, ["工事番号", "案件番号", "管理番号"]);
 const colName = findColMulti(mHead, ["工事名称", "案件名", "工事名"]);
  const list = [];
 for (let i = 1; i < mData.length; i++) {
   const mId = String(mData[i][0]);
   const mStatus = colStatus !== -1 ? String(mData[i][colStatus]) : "";
   const mKubun = colKubun !== -1 ? String(mData[i][colKubun]) : "一般";
   const mNo = colNo !== -1 ? String(mData[i][colNo]) : String(mData[i][1] || "-");
   const mName = colName !== -1 ? String(mData[i][colName]) : String(mData[i][2] || mNo);

   if (existingIds.has(mId) && ["着工前", "着工中", "完工"].includes(mStatus)) {
     list.push({ id: mId, no: mNo, name: mName, status: mStatus, kubun: mKubun });
   }
 }
 return list;
}

function getAggregatedData(startDateStr, endDateStr, targetProjectId = "", isAllTime = false) {
 try {
   const start = isAllTime ? new Date("2000/01/01") : new Date(startDateStr);
   const end = new Date(endDateStr);
   end.setHours(23, 59, 59, 999);

   const masterSS = SpreadsheetApp.openById(SS_MASTER_ID);
   const resourceSS = SpreadsheetApp.openById(SS_RESOURCES_ID);
  
   let nippoSht = resourceSS.getSheetByName("工事日報") || resourceSS.getSheetByName("業務予定");
   let masterSht = masterSS.getSheetByName("工事案件マスター") || masterSS.getSheetByName("案件マスタ");
   let matSht = resourceSS.getSheetByName("使用材料") || resourceSS.getSheetByName("使用材料リスト") || resourceSS.getSheetByName("材料");
   let partnerSht = resourceSS.getSheetByName("日報_協力業者") || resourceSS.getSheetByName("業務予定_協力業者");

   const nippoData = nippoSht.getDataRange().getValues();
   const masterData = masterSht.getDataRange().getValues();
   const matData = matSht ? matSht.getDataRange().getValues() : [];
   const matRichData = matSht ? matSht.getDataRange().getRichTextValues() : [];
   const ptData = partnerSht ? partnerSht.getDataRange().getValues() : [];

   const nHead = nippoData[0];
   const colDate = findColMulti(nHead, ["報告日時", "業務日", "日付"]);
   const colProjId = findColMulti(nHead, ["工事案件", "工事案件ID", "案件ID"]);
   const colStaff = findColMulti(nHead, ["HITEC作業員", "作業員", "氏名"]);
   const colNippoId = findColMulti(nHead, ["日報ID", "業務予定ID", "ID"]);
   const colWorkKubun = findColMulti(nHead, ["作業区分", "区分"]);
   const colStartTime = findColMulti(nHead, ["作業開始時間", "開始時間", "出社時間", "開始"]);
   const colEndTime = findColMulti(nHead, ["作業終了時間", "終了時間", "退社時間", "終了"]);
  
   const colCar = findColMulti(nHead, ["作業車", "車両"]);
   const colMachine = findColMulti(nHead, ["建設機械", "建機", "重機"]);

   const driveFileCache = {};
   const getDriveLink = (fileName, folderId) => {
     if (!fileName || !folderId) return null;
     const cleanName = fileName.split('/').pop().trim();
     if (!cleanName) return null;
     if (driveFileCache[cleanName]) return driveFileCache[cleanName];
     try {
       const folder = DriveApp.getFolderById(folderId);
       const escapedName = cleanName.replace(/'/g, "\\'");
       const files = folder.searchFiles(`title = '${escapedName}'`);
       if (files.hasNext()) {
         const file = files.next();
         const url = file.getUrl();
         driveFileCache[cleanName] = url;
         return url;
       }
     } catch(e) {}
     return null;
   };

   const extractLinksWithDrive = (textVal, richTextVal, folderId) => {
     const results = [];
     const addResult = (url, name) => {
       if (!results.find(r => r.url === url)) results.push({ url: url, fileName: name });
     };
     if (richTextVal && typeof richTextVal.getRuns === 'function') {
       richTextVal.getRuns().forEach(run => {
         const url = run.getLinkUrl();
         if (url) addResult(url, run.getText() || "リンク");
       });
     }
     const textItems = String(textVal || "").split(/[\n,，、\s]+/).map(v => v.trim()).filter(v => v !== "");
     textItems.forEach(v => {
       const lower = v.toLowerCase();
       if (lower.startsWith("http")) {
         addResult(v, v.split('/').pop() || "リンク");
       } else if (/\.(jpg|jpeg|png|gif|pdf|xlsx|docx|zip)$/.test(lower) || lower.includes('/')) {
         const cleanName = v.split('/').pop();
         const driveUrl = getDriveLink(v, folderId);
         if (driveUrl) addResult(driveUrl, cleanName);
         else addResult(v, cleanName);
       }
     });
     return results;
   };

   const materialMap = {};
   if (matData.length > 0) {
     const mHead = matData[0];
     const colMRefId = findColMulti(mHead, ["日報ID", "業務予定ID", "ID"]);
     const colMName = findColMulti(mHead, ["材料名", "使用材料", "品名", "材料"]);
     const colMPhoto = findColMulti(mHead, ["現場写真", "写真", "画像"]);
     const colMDoc = findColMulti(mHead, ["資料・図面", "資料", "図面", "ファイル"]);
    
     if (colMRefId !== -1) {
       for (let k = 1; k < matData.length; k++) {
         const rid = String(matData[k][colMRefId]);
         if (!materialMap[rid]) materialMap[rid] = { materials: [], photos: [], docs: [] };
         if (colMName !== -1) {
           const mname = String(matData[k][colMName]);
           if (mname && mname !== "undefined" && mname.trim() !== "") materialMap[rid].materials.push(mname.trim());
         }
         if (colMPhoto !== -1) materialMap[rid].photos.push(...extractLinksWithDrive(matData[k][colMPhoto], matRichData[k][colMPhoto], PHOTO_FOLDER_ID));
         if (colMDoc !== -1) materialMap[rid].docs.push(...extractLinksWithDrive(matData[k][colMDoc], matRichData[k][colMDoc], DOC_FOLDER_ID));
       }
     }
   }

   const projMasterMap = {};
   const mHead = masterData[0];
   const mNoIdx = findColMulti(mHead, ["工事番号", "案件番号"]) !== -1 ? findColMulti(mHead, ["工事番号", "案件番号"]) : 1;
   const mNameIdx = findColMulti(mHead, ["工事名称", "案件名"]) !== -1 ? findColMulti(mHead, ["工事名称", "案件名"]) : 2;
   const mKubunIdx = findColMulti(mHead, ["区分"]) !== -1 ? findColMulti(mHead, ["区分"]) : 5;

   for(let i=1; i<masterData.length; i++){
     const row = masterData[i];
     const mId = String(row[0]);
     const mkubun = String(row[mKubunIdx]);
     let dName = row[mNameIdx];
     if (["一般", "役所"].includes(mkubun)) dName = `${row[3]}：${row[mNameIdx]}`;
     else if (["川北", "BPE"].includes(mkubun)) dName = `${row[4]}：${row[mNameIdx]}`;
     projMasterMap[mId] = { name: dName, kubun: mkubun, no: row[mNoIdx] };
   }

   const partnerMap = {};
   if (ptData.length > 0) {
     const ptHead = ptData[0];
     const colPtRefId = findColMulti(ptHead, ["日報ID", "業務予定ID", "ID"]);
     const colPtName = findColMulti(ptHead, ["業者名", "協力業者"]);
     const colPtCount = findColMulti(ptHead, ["人数", "作業員数"]);
     if (colPtRefId !== -1) {
       for (let j = 1; j < ptData.length; j++) {
         const rid = String(ptData[j][colPtRefId]);
         if (!partnerMap[rid]) partnerMap[rid] = [];
         partnerMap[rid].push({ name: String(ptData[j][colPtName]), count: Number(ptData[j][colPtCount]) || 0 });
       }
     }
   }

   const results = {
     projects: {}, staff: {}, companies: {},
     summary: {
       totalPeople: 0, totalHours: 0, totalOT: 0, equipment: {},
       kubunTotals: { kouji: 0, kanri: 0, mitsumori: 0 },
       kubunDetails: { kouji: {normal:0, ot:0}, kanri: {normal:0, ot:0}, mitsumori: {normal:0, ot:0} }
     }
   };

   for (let i = 1; i < nippoData.length; i++) {
     const row = nippoData[i];
     const rDate = new Date(row[colDate]);
     if (isNaN(rDate.getTime()) || rDate < start || rDate > end) continue;

     const pId = String(row[colProjId]);
     if (targetProjectId && pId !== targetProjectId) continue;

     const nId = String(row[colNippoId]);
     const wKubun = String(row[colWorkKubun] || "");
     const sStr = String(row[colStaff] || "");
     const staffsRaw = sStr.split(/[｜|，,、]+/).map(s => s.trim()).filter(s => s !== "");
    
     const timeInfo = calculateActualHours(row[colStartTime], row[colEndTime]);
     const totalH = timeInfo.normal + timeInfo.ot;

     // ｜ や | で区切られた車両・建機を別々にカウントする
     const rawCar = colCar !== -1 ? String(row[colCar] || "").split(/[｜|，,、\s]+/).map(s => s.trim()).filter(s => s !== "") : [];
     const rawMachine = colMachine !== -1 ? String(row[colMachine] || "").split(/[｜|，,、\s]+/).map(s => s.trim()).filter(s => s !== "") : [];

     let cat = "mitsumori";
     if (wKubun.includes("工事")) cat = "kouji";
     else if (wKubun.includes("管理")) cat = "kanri";

     if (!results.projects[pId]) {
       const info = projMasterMap[pId] || { name: `不明(${pId})`, kubun: "一般", no: "-" };
       results.projects[pId] = {
         id: pId, no: info.no, name: info.name, kubun: info.kubun,
         staffCount: 0, partnerCount: 0,
         normalHours: 0, overtimeHours: 0, totalHours: 0,
         breakdown: { kouji: 0, kanri: 0, mitsumori: 0 },
         breakdownDetails: { kouji: {normal:0, ot:0}, kanri: {normal:0, ot:0}, mitsumori: {normal:0, ot:0} },
         materials: [], photos: [], docs: [],
         equipment: {},
         dailyLogs: [] 
       };
     }

     const pObj = results.projects[pId];

     if (materialMap[nId]) {
       pObj.materials.push(...materialMap[nId].materials);
       pObj.photos.push(...materialMap[nId].photos);
       pObj.docs.push(...materialMap[nId].docs);
     }

     const partners = partnerMap[nId] || [];
     partners.forEach(p => {
       pObj.partnerCount += p.count;
       results.summary.totalPeople += p.count;
       if (!results.companies[p.name]) results.companies[p.name] = 0;
       results.companies[p.name] += p.count;
     });

     pObj.dailyLogs.push({
       date: Utilities.formatDate(rDate, "JST", "yyyy/MM/dd"),
       kubun: wKubun || "一般",
       staffs: staffsRaw.join(", "),
       partners: partners.map(p => p.name).join(", "),
       hours: totalH,
       ot: timeInfo.ot,
       car: rawCar.join(", "),
       machine: rawMachine.join(", ")
     });

     const eqList = [...rawCar, ...rawMachine];
     eqList.forEach(eq => {
       pObj.equipment[eq] = (pObj.equipment[eq] || 0) + 1;
       results.summary.equipment[eq] = (results.summary.equipment[eq] || 0) + 1;
     });

     staffsRaw.forEach(rawName => {
       const nKey = rawName.replace(/[\s　]+/g, "");
       if (!nKey) return;
       const dName = rawName.replace(/[\s　]+/g, " ");

       // 案件集計
       pObj.staffCount++;
       pObj.normalHours += timeInfo.normal;
       pObj.overtimeHours += timeInfo.ot;
       pObj.totalHours += totalH;
       pObj.breakdown[cat] += totalH;
       pObj.breakdownDetails[cat].normal += timeInfo.normal;
       pObj.breakdownDetails[cat].ot += timeInfo.ot;

       // 全体サマリー集計
       results.summary.totalPeople++;
       results.summary.totalHours += timeInfo.normal;
       results.summary.totalOT += timeInfo.ot;
       results.summary.kubunTotals[cat] += totalH;
       results.summary.kubunDetails[cat].normal += timeInfo.normal;
       results.summary.kubunDetails[cat].ot += timeInfo.ot;

       // 作業員別集計
       if (!results.staff[nKey]) {
         results.staff[nKey] = { displayName: dName, kouji: { normal: 0, ot: 0 }, kanri: { normal: 0, ot: 0 }, mitsumori: { normal: 0, ot: 0 } };
       }
       results.staff[nKey][cat].normal += timeInfo.normal;
       results.staff[nKey][cat].ot += timeInfo.ot;
     });
   }

   Object.values(results.projects).forEach(p => {
     p.materials = [...new Set(p.materials)];
     p.photos = p.photos.filter((obj, index, self) => index === self.findIndex((t) => t.url === obj.url));
     p.docs = p.docs.filter((obj, index, self) => index === self.findIndex((t) => t.url === obj.url));
     p.dailyLogs.sort((a, b) => new Date(a.date) - new Date(b.date));
   });

   return results;

 } catch (e) {
   return { error: "サーバー側エラー: " + e.toString() };
 }
}

function calculateActualHours(startVal, endVal) {
 const parseTime = (val) => {
   if (val == null || val === '') return null;
   if (val instanceof Date) return val;
   if (typeof val === 'string') {
     const match = val.match(/(\d{1,2})[:：](\d{1,2})/);
     if (match) {
       const d = new Date();
       d.setHours(parseInt(match[1]), parseInt(match[2]), 0, 0);
       return d;
     }
   }
   if (typeof val === 'number') {
     const d = new Date();
     const totalMinutes = Math.round(val * 24 * 60);
     d.setHours(Math.floor(totalMinutes / 60), totalMinutes % 60, 0, 0);
     return d;
   }
   return null;
 };
  const st = parseTime(startVal);
 const et = parseTime(endVal);
  if (!st || !et) return { normal: 0, ot: 0 };
 const startMin = st.getHours() * 60 + st.getMinutes();
 const endMin = et.getHours() * 60 + et.getMinutes();
 if (endMin <= startMin) return { normal: 0, ot: 0 };

 const standardLimit = 17 * 60;
 const breakStart = 12 * 60;
 const breakEnd = 13 * 60;

 let normal = Math.min(standardLimit, endMin) - startMin;
  if (startMin <= breakStart && endMin >= breakEnd) normal -= 60;
 else if (startMin > breakStart && startMin < breakEnd) normal -= (breakEnd - startMin);
 else if (endMin > breakStart && endMin < breakEnd) normal -= (endMin - breakStart);

 let ot = Math.max(0, endMin - Math.max(startMin, standardLimit));
 return { normal: Math.max(0, normal / 60), ot: Math.max(0, ot / 60) };
}

   let allProjects = [];

   window.onload = () => {
     const now = new Date();
     document.getElementById('dateStart').value = new Date(now.getFullYear(), now.getMonth(), 1).toISOString().split('T')[0];
     document.getElementById('dateEnd').value = now.toISOString().split('T')[0];
    
     google.script.run
       .withSuccessHandler(list => {
         allProjects = list.sort((a, b) => b.no.localeCompare(a.no, undefined, {numeric: true}));
         updateProjectDropdown();
       })
       .withFailureHandler(err => alert("案件リスト取得エラー:" + err))
       .getProjectList();
   };

   function updateProjectDropdown() {
     const kubunVal = document.getElementById('kubunFilter').value;
     const select = document.getElementById('projectSelect');
     select.innerHTML = '<option value="">全ての案件を表示</option>';
     const filtered = allProjects.filter(p => (kubunVal === "ALL" || p.kubun === kubunVal));
     filtered.forEach(p => {
       const opt = document.createElement('option');
       opt.value = p.id;
       const noDisplay = p.no && p.no !== "-" ? `[${p.no}]` : "[番号未設定]";
       opt.innerText = `${noDisplay} ${p.name} (${p.status})`;
       select.appendChild(opt);
     });
   }

   function loadReport() {
     const start = document.getElementById('dateStart').value;
     const end = document.getElementById('dateEnd').value;
     const projectId = document.getElementById('projectSelect').value;
     const isAllTime = document.getElementById('allTimeCheck').checked;
    
     if (!start || !end) {
       alert("期間を設定してください");
       return;
     }
     document.getElementById('loading').style.display = 'flex';
     google.script.run
       .withSuccessHandler(data => {
         document.getElementById('loading').style.display = 'none';
         render(data);
       })
       .withFailureHandler(err => {
         document.getElementById('loading').style.display = 'none';
         alert("集計中にエラーが発生しました: " + err);
       })
       .getAggregatedData(start, end, projectId, isAllTime);

            const [invoices, setInvoices] = useState([]);
            const [details, setDetails] = useState([]);
            const [isLoading, setIsLoading] = useState(true);

            // アコーディオンの開閉状態を管理するState
            const [expandedInvoices, setExpandedInvoices] = useState({});
            const toggleInvoice = (id) => {
                setExpandedInvoices(prev => ({...prev, [id]: !prev[id]}));
            };

            // モーダル
            const [isModalOpen, setIsModalOpen] = useState(false);
            const [isEditing, setIsEditing] = useState(false);
            const [selectedProject, setSelectedProject] = useState(null);
            const [editingDetailId, setEditingDetailId] = useState(null);
            const [isSaving, setIsSaving] = useState(false);
            const [formData, setFormData] = useState({
                type: '出来高', title: '', clientName: '', contractAmount: '', month: '', amount: '', billDate: '', payDate: '', memo: '',
                projectIds: [] 
            });

            // データの読み込み
            const loadAllData = useCallback(() => {
                setIsLoading(true);
                if (typeof google === 'undefined' || !google.script) {
                    setIsLoading(false); return;
                }

                google.script.run.withSuccessHandler((role) => {
                    if (role) setUserRole(role);
                }).getUserRole();

                google.script.run.withSuccessHandler((pData) => {
                    const parsedP = JSON.parse(pData).map(row => ({
                        id: String(row['管理ID'] || ''), no: row['工事番号'] || '-', name: row['工事名称'] || '-', 
                        client: row['発注者'] || '-', category: row['区分'] || '', siteName: row['現場名'] || '', status: row['工事状態フラグ'] || ''
                    }));
                    setProjects(parsedP);
                    google.script.run.withSuccessHandler((bData) => {
                        const parsedB = JSON.parse(bData);
                        setInvoices((parsedB.invoices || []).map(row => ({
                            id: String(row['ID'] || row.id || ''), 
                            projectId: String(row['対象工事リスト'] || row['対象工事ID'] || row.projectId || ''),
                            type: row['請求区分'] || row['区分'] || row.type || '出来高', 
                            title: row['請求件名'] || row.title || '-',
                            clientName: row['請求先名称'] || row.clientName || '-', 
                            contractAmount: Number(row['請負金額'] || row.contractAmount || 0)
                        })));
                        setDetails((parsedB.details || []).map(row => ({
                            id: String(row['ID'] || row.id || ''), 
                            invoiceId: String(row['請求書ID'] || row.invoiceId || ''),
                            month: row['請求対象月'] || row['対象月'] || row.month || '', 
                            amount: Number(row['金額'] || row.amount || 0),
                            billDate: row['請求日'] || row.billDate || '', 
                            payDate: row['入金予定日'] || row.payDate || '',
                            nyukinbi: row['入金日'] || row.nyukinbi || '', 
                            status: row['明細ステータス'] || row['ステータス'] || row.status || '請求済み', 
                            memo: row['明細備考'] || row['備考'] || row.memo || ''
                        })));
                        setIsLoading(false);
                    }).getBillingData();
                }).getProjectsData();
            }, []);

            useEffect(() => { loadAllData(); }, [loadAllData]);

            const isViewer = userRole === 'viewer';

            const getRelatedInvoices = (projectId, invList) => {
                if (!projectId) return [];
                return invList.filter(inv => {
                    if (!inv.projectId || inv.projectId === 'none') return false;
                    const ids = String(inv.projectId).split(',').map(s => s.trim());
                    return ids.includes(String(projectId));
                });
            };

            const projectsWithData = useMemo(() => {
                return projects.map(p => {
                    const relatedInvoices = getRelatedInvoices(p.id, invoices);
                    const relatedDetails = details.filter(det => relatedInvoices.some(inv => inv.id === det.invoiceId));
                    const hasCompleted = relatedInvoices.some(inv => inv.type === '完成');
                    const hasPartial = relatedInvoices.some(inv => inv.type === '出来高');
                    const allPaid = relatedDetails.length > 0 && relatedDetails.every(d => d.status === '入金済み');
                    return { ...p, hasCompleted, hasPartial, allPaid };
                });
            }, [projects, invoices, details]);

            // 【案件一覧タブ】
            const filteredProjects = useMemo(() => {
                let list = projectsWithData.filter(p => p.status === '着工中' || p.status === '完工');
                list = list.filter(p => !p.hasCompleted);
                
                if (projectFilter === '着工中') list = list.filter(p => p.status === '着工中' && !p.hasPartial);
                if (projectFilter === '出来高発行済') list = list.filter(p => p.hasPartial);
                if (projectFilter === '完工') list = list.filter(p => p.status === '完工');

                if (searchQuery) {
                    const q = searchQuery.toLowerCase();
                    list = list.filter(p => 
                        String(p.name || '').toLowerCase().includes(q) || 
                        String(p.no || '').toLowerCase().includes(q) || 
                        String(p.client || '').toLowerCase().includes(q)
                    );
                }
                return list;
            }, [projectsWithData, projectFilter, searchQuery]);

            // 【請求中タブ】
            const billingQueue = useMemo(() => {
                let queue = invoices.map(inv => {
                    const invDetails = details.filter(d => d.invoiceId === inv.id);
                    
                    const projectIds = inv.projectId && inv.projectId !== 'none' 
                        ? String(inv.projectId).split(',').map(s => s.trim()) : [];
                    const relatedProjs = projects.filter(p => projectIds.includes(String(p.id)));
                    
                    const pendingDetails = invDetails.filter(d => d.status === '請求済み'); 
                    const isCompleted = inv.type === '完成' || 
                                        (relatedProjs.length > 0 && relatedProjs.every(p => p.status === '完工'));
                    
                    if (pendingDetails.length === 0 && isCompleted) return null;
                    if (invDetails.length === 0) return null; 

                    const totalBilled = invDetails.reduce((sum, d) => sum + (Number(d.amount) || 0), 0);
                    const projectNos = relatedProjs.length > 0 ? relatedProjs.map(p => p.no).join(', ') : '案件外';

                    // いずれかの明細で入金予定日を過ぎているかチェック
                    const hasOverdue = pendingDetails.some(d => isDateOverdue(d.payDate));

                    return { 
                        ...inv, 
                        pendingDetails, 
                        totalBilled, 
                        balance: (Number(inv.contractAmount) || 0) - totalBilled, 
                        projectNo: projectNos,
                        displayProjectId: inv.projectId !== 'none' ? inv.projectId : '案件外',
                        relatedProjects: relatedProjs, // ★関連するすべての案件データを保持
                        hasOverdue
                    };
                }).filter(x => x !== null);

                if (searchQuery) {
                    const q = searchQuery.toLowerCase();
                    queue = queue.filter(inv => 
                        String(inv.title || '').toLowerCase().includes(q) || 
                        String(inv.projectNo || '').toLowerCase().includes(q) || 
                        String(inv.projectId || '').toLowerCase().includes(q) ||
                        String(inv.clientName || '').toLowerCase().includes(q)
                    );
                }
                return queue;
            }, [invoices, details, projects, searchQuery]);

            // 【入金履歴タブ】完了した請求書リスト（請求書ごとにグループ化）
            const completedInvoices = useMemo(() => {
                let list = invoices.map(inv => {
                    const invDetails = details.filter(d => d.invoiceId === inv.id);
                    const paidDetails = invDetails.filter(d => d.status === '入金済み');
                    const pendingDetails = invDetails.filter(d => d.status === '請求済み');
                    
                    const projectIds = inv.projectId && inv.projectId !== 'none' 
                        ? String(inv.projectId).split(',').map(s => s.trim()) : [];
                    const relatedProjs = projects.filter(p => projectIds.includes(String(p.id)));
                    
                    const isCompleted = inv.type === '完成' || 
                                        (relatedProjs.length > 0 && relatedProjs.every(p => p.status === '完工'));
                    
                    if (!isCompleted || pendingDetails.length > 0 || paidDetails.length === 0) return null;

                    const totalBilled = paidDetails.reduce((sum, d) => sum + (Number(d.amount) || 0), 0);
                    const projectNos = relatedProjs.length > 0 ? relatedProjs.map(p => p.no).join(', ') : '案件外';

                    const lastPayDate = paidDetails.reduce((latest, d) => {
                        const dStr = d.nyukinbi || d.billDate;
                        if (!dStr) return latest;
                        const dDate = new Date(dStr);
                        if (isNaN(dDate.getTime())) return latest;
                        return dDate > latest ? dDate : latest;
                    }, new Date(0));

                    return { 
                        ...inv, 
                        paidDetails, 
                        totalBilled, 
                        projectNo: projectNos,
                        displayProjectId: inv.projectId !== 'none' ? inv.projectId : '案件外',
                        relatedProjects: relatedProjs, // ★関連するすべての案件データを保持
                        lastPayDate
                    };
                }).filter(x => x !== null);

                if (searchQuery) {
                    const q = searchQuery.toLowerCase();
                    list = list.filter(inv => 
                        String(inv.title || '').toLowerCase().includes(q) || 
                        String(inv.projectNo || '').toLowerCase().includes(q) || 
                        String(inv.displayProjectId || '').toLowerCase().includes(q) ||
                        String(inv.clientName || '').toLowerCase().includes(q)
                    );
                }
                
                return list.sort((a, b) => b.lastPayDate - a.lastPayDate);
            }, [invoices, details, projects, searchQuery]);

            const baseClient = selectedProject ? getDisplayClient(selectedProject) : null;
            
            // ★修正: 合算選択用リスト。既に選択されているものは状態に関わらず表示する。
            const availableProjects = useMemo(() => {
                if (!baseClient) return [];
                return projectsWithData.filter(p => {
                    // メインのプロジェクトは上部に表示されるため、下の追加リストからは除外
                    if (selectedProject && p.id === selectedProject.id) return false;
                    
                    // すでに合算対象として選ばれている案件は無条件でリストに表示する（チェックを外せるように）
                    if (formData.projectIds.includes(p.id)) return true;

                    // それ以外の追加候補の条件
                    if (getDisplayClient(p) !== baseClient) return false;
                    if (p.status !== '着工中' && p.status !== '完工') return false;
                    if (p.hasCompleted) return false;
                    return true;
                });
            }, [projectsWithData, baseClient, selectedProject, formData.projectIds]);

            const handleCancelPartial = (projectId) => {
                if (isViewer) return;
                const targetInvoices = invoices.filter(inv => {
                    if (!inv.projectId || inv.projectId === 'none') return false;
                    return String(inv.projectId).split(',').map(s => s.trim()).includes(String(projectId)) && inv.type === '出来高';
                });

                if (targetInvoices.length === 0) {
                    alert('対象の請求書データが見つかりません。');
                    return;
                }

                const hasDetails = targetInvoices.some(inv => details.some(d => String(d.invoiceId) === String(inv.id)));
                if (hasDetails) {
                    alert('【取消できません】\nすでに請求明細が発行されています。\n\n状態を元に戻すには、まず「請求中・入金待ち」または「入金完了履歴」タブから、この案件の発行済明細（ゴミ箱マーク）をすべて削除してください。');
                    return;
                }

                if (!confirm('この案件の「出来高請求中」状態を取り消して、未請求に戻しますか？（合算されている場合、他の案件も同時に未請求に戻ります）')) return;

                const invoiceIdToDelete = targetInvoices[0].id;

                if (typeof google !== 'undefined' && google.script) {
                    google.script.run
                        .withSuccessHandler(() => {
                            alert('出来高請求状態を取り消しました。');
                            loadAllData();
                        })
                        .withFailureHandler((err) => {
                            alert("取消に失敗しました。\nGAS側のコード更新（新バージョンでのデプロイ）がされていない可能性があります。\n\nエラー詳細: " + err.message);
                        })
                        .deleteInvoiceParent(invoiceIdToDelete);
                } else {
                    setInvoices(invoices.filter(inv => inv.id !== invoiceIdToDelete));
                    alert('出来高請求状態を取り消しました。(ローカル表示)');
                }
            };

            const handleOpenModal = (project, existingDetail = null) => {
                if (isViewer) return;
                const today = new Date().toISOString().split('T')[0];
                
                const toInputDate = (dStr) => {
                    if (!dStr) return '';
                    const d = new Date(dStr);
                    if (isNaN(d.getTime())) return String(dStr).split('T')[0];
                    return `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}-${String(d.getDate()).padStart(2, '0')}`;
                };

                if (existingDetail) {
                    setIsEditing(true);
                    setEditingDetailId(existingDetail.id);
                    const inv = invoices.find(i => i.id === existingDetail.invoiceId);
                    
                    let safeMonth = existingDetail.month || '';
                    if (safeMonth) {
                        if (String(safeMonth).includes('T')) {
                            const d = new Date(safeMonth);
                            if (!isNaN(d.getTime())) {
                                safeMonth = `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}`;
                            }
                        } else if (safeMonth.length > 7) {
                            safeMonth = safeMonth.substring(0, 7);
                        }
                    }

                    let initialProjectIds = inv?.projectId && inv.projectId !== 'none' 
                        ? inv.projectId.split(',').map(s => s.trim()) : [];

                    setFormData({
                        type: inv?.type || '出来高', title: inv?.title || '', clientName: inv?.clientName || '',
                        contractAmount: inv?.contractAmount || 0, month: safeMonth,
                        amount: existingDetail.amount || '', 
                        billDate: toInputDate(existingDetail.billDate),
                        payDate: toInputDate(existingDetail.payDate), 
                        memo: existingDetail.memo || '',
                        projectIds: initialProjectIds
                    });

                    const mainProject = projects.find(p => p.id === initialProjectIds[0]) || null;
                    setSelectedProject(mainProject);
                } else {
                    setIsEditing(false);
                    let client = getDisplayClient(project);

                    const existingInv = project ? invoices.find(inv => {
                        if (!inv.projectId || inv.projectId === 'none') return false;
                        return String(inv.projectId).split(',').map(s => s.trim()).includes(String(project.id));
                    }) : null;
                    
                    let initialProjectIds = existingInv?.projectId && existingInv.projectId !== 'none' 
                        ? existingInv.projectId.split(',').map(s => s.trim()) 
                        : (project ? [project.id] : []);

                    setFormData({
                        type: project === null ? '完成' : (project?.status === '完工' ? '完成' : '出来高'),
                        title: existingInv ? existingInv.title : (project?.name || ''),
                        clientName: existingInv ? existingInv.clientName : client,
                        contractAmount: existingInv ? existingInv.contractAmount : '',
                        month: today.substring(0, 7), amount: '', billDate: today, payDate: '', memo: '',
                        projectIds: initialProjectIds
                    });
                    setSelectedProject(project);
                }
                setIsModalOpen(true);
            };

            const handleSave = () => {
                if (isViewer) return;
                if (!formData.amount || !formData.billDate) return alert('金額と請求日は必須です。');
                if (formData.type === '出来高' && !formData.month) return alert('出来高請求の場合は「対象月」を選択してください。');
                
                setIsSaving(true);

                const combinedProjectIds = formData.projectIds.length > 0 ? formData.projectIds.join(', ') : 'none';

                if (isEditing) {
                    const updatedFormData = { ...formData, projectIdsArray: formData.projectIds };
                    if (google?.script) {
                        google.script.run.withSuccessHandler(() => {
                            loadAllData(); setIsSaving(false); setIsModalOpen(false);
                        }).updateDetailField(editingDetailId, 'full_update', JSON.stringify(updatedFormData));
                    } else {
                        setTimeout(() => { loadAllData(); setIsSaving(false); setIsModalOpen(false); }, 500);
                    }
                } else {
                    const existingInv = selectedProject ? invoices.find(inv => {
                        if (!inv.projectId || inv.projectId === 'none') return false;
                        return String(inv.projectId).split(',').map(s => s.trim()).includes(String(selectedProject.id));
                    }) : null;

                    const invObj = existingInv || {
                        id: 'inv_' + Math.random().toString(36).substr(2, 9),
                        projectId: combinedProjectIds,
                        type: formData.type, title: formData.title, clientName: formData.clientName, contractAmount: Number(formData.contractAmount) || 0
                    };
                    const detObj = {
                        id: 'det_' + Math.random().toString(36).substr(2, 9), invoiceId: invObj.id,
                        month: formData.month || '',
                        amount: Number(formData.amount), billDate: formData.billDate, payDate: formData.payDate,
                        nyukinbi: '', status: '請求済み', memo: formData.memo
                    };
                    
                    if (google?.script) {
                        google.script.run.withSuccessHandler(() => {
                            if (existingInv && existingInv.type !== formData.type) {
                                const existingDet = details.find(d => d.invoiceId === existingInv.id);
                                if (existingDet) {
                                    const toInputDate = (dStr) => {
                                        if (!dStr) return '';
                                        const d = new Date(dStr);
                                        if (isNaN(d.getTime())) return String(dStr).split('T')[0];
                                        return `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}-${String(d.getDate()).padStart(2, '0')}`;
                                    };
                                    let safeMonth = existingDet.month || '';
                                    if (safeMonth) {
                                        if (String(safeMonth).includes('T')) {
                                            const d = new Date(safeMonth);
                                            if (!isNaN(d.getTime())) safeMonth = `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}`;
                                        } else if (safeMonth.length > 7) {
                                            safeMonth = safeMonth.substring(0, 7);
                                        }
                                    }
                                    const updatedFormData = {
                                        type: formData.type,
                                        title: formData.title,
                                        clientName: formData.clientName,
                                        contractAmount: formData.contractAmount,
                                        projectIdsArray: formData.projectIds,
                                        month: safeMonth,
                                        amount: existingDet.amount,
                                        billDate: toInputDate(existingDet.billDate),
                                        payDate: toInputDate(existingDet.payDate),
                                        memo: existingDet.memo
                                    };
                                    google.script.run.withSuccessHandler(() => {
                                        loadAllData(); setIsSaving(false); setIsModalOpen(false); setActiveTab('invoices');
                                    }).updateDetailField(existingDet.id, 'full_update', JSON.stringify(updatedFormData));
                                    return; 
                                }
                            }
                            loadAllData(); setIsSaving(false); setIsModalOpen(false); setActiveTab('invoices');
                        }).saveNewInvoice(invObj, detObj);
                    } else {
                        setTimeout(() => { loadAllData(); setIsSaving(false); setIsModalOpen(false); setActiveTab('invoices'); }, 500);
                    }
                }
            };

            const handleStatusToggle = (id, isPaid) => {
                if (isViewer) return;
                if (google?.script) google.script.run.withSuccessHandler(() => loadAllData()).updatePaidStatus(id, isPaid);
                else setDetails(details.map(d => d.id === id ? { ...d, status: isPaid ? '入金済み' : '請求済み' } : d));
            };

            const handleDelete = (id) => {
                if (isViewer) return;
                if (!confirm('この明細データを完全に削除しますか？')) return;
                if (google?.script) google.script.run.withSuccessHandler(() => loadAllData()).deleteInvoiceData(id);
                else setDetails(details.filter(d => d.id !== id));
            };

            const handleFieldUpdate = (id, field, value) => {
                if (isViewer) return;
                setDetails(details.map(d => d.id === id ? { ...d, [field]: value } : d));
                if (google?.script) google.script.run.updateDetailField(id, field, value);
            };

            return (
                <div className="min-h-screen pb-20 text-sm md:text-base">
                    {/* Header */}
                    <header className="bg-white border-b-4 border-blue-600 sticky top-0 z-20 shadow-sm">
                        <div className="max-w-6xl mx-auto px-4">
                            <div className="flex flex-col md:flex-row justify-between items-center h-auto md:h-16 py-3 md:py-0 gap-3">
                                <div className="flex items-center gap-3">
                                    <img src="./logo_1A.png" alt="HITEC" className="h-6 md:h-8 object-contain" onError={(e) => { e.target.style.display = 'none'; }} />
                                    <h1 className="text-xl font-black text-blue-800 flex items-center gap-2">
                                        <Icons.Receipt /> 建設DX 請求管理
                                    </h1>
                                    {isViewer && (
                                        <span className="ml-2 px-3 py-1 bg-slate-200 text-slate-600 text-xs font-black rounded-lg shadow-inner">
                                            👁️ 閲覧モード
                                        </span>
                                    )}
                                </div>
                                {!isViewer && (
                                    <button onClick={() => handleOpenModal(null)} className="flex items-center gap-1.5 px-4 py-2 bg-orange-500 text-white font-bold text-sm rounded-xl hover:bg-orange-600 shadow-sm transition-all active:scale-95 whitespace-nowrap">
                                        <Icons.Plus /> 案件外で新規作成
                                    </button>
                                )}
                            </div>
                            <div className="flex space-x-1 overflow-x-auto no-scrollbar">
                                {[
                                    { id: 'projects', label: '案件一覧', icon: <Icons.Folder /> },
                                    { id: 'invoices', label: '請求中・入金待ち', icon: <Icons.Receipt /> },
                                    { id: 'history', label: '入金完了履歴', icon: <Icons.History /> }
                                ].map(tab => (
                                    <button key={tab.id} onClick={() => setActiveTab(tab.id)} className={`flex items-center gap-2 px-5 py-3 font-bold text-sm border-b-4 transition-all whitespace-nowrap ${activeTab === tab.id ? 'border-blue-600 text-blue-600 bg-blue-50' : 'border-transparent text-slate-400 hover:bg-slate-50'}`}>
                                        {tab.icon} {tab.label}
                                    </button>
                                ))}
                            </div>
                        </div>
                    </header>

                    <main className="max-w-6xl mx-auto px-4 py-8">
                        {/* Search Bar */}
                        <div className="mb-6 relative max-w-xl">
                            <div className="absolute left-4 top-1/2 -translate-y-1/2 text-slate-400"><Icons.Search /></div>
                            <input type="text" placeholder="案件名、工事番号(リスト)、発注者で検索..." value={searchQuery} onChange={e => setSearchQuery(e.target.value)} className="w-full pl-12 pr-4 py-3 bg-white border-2 border-slate-200 rounded-2xl text-sm font-bold shadow-sm focus:border-blue-500 outline-none transition-all" />
                        </div>

                        {isLoading ? (
                            <div className="flex flex-col items-center justify-center py-24 bg-white rounded-2xl shadow-sm border border-slate-200">
                                <div className="text-blue-600 mb-4 scale-150"><Icons.Loader /></div>
                                <div className="text-slate-500 font-bold tracking-widest animate-pulse text-sm uppercase">データを同期中...</div>
                            </div>
                        ) : activeTab === 'projects' ? (
                            <div className="bg-white rounded-2xl shadow-sm overflow-hidden border border-slate-200">
                                <div className="p-5 bg-slate-50 border-b border-slate-200 flex flex-col md:flex-row justify-between items-center gap-4">
                                    <div className="flex gap-2 overflow-x-auto w-full no-scrollbar pb-2 md:pb-0">
                                        {[
                                            { id: '着工中', label: '着工中 (未請求)' },
                                            { id: '完工', label: '完工 (未請求)' },
                                            { id: '出来高発行済', label: '出来高請求中' }
                                        ].map(f => (
                                            <button key={f.id} onClick={() => setProjectFilter(f.id)} className={`px-4 py-2 rounded-xl text-sm font-bold whitespace-nowrap transition-all ${projectFilter === f.id ? 'bg-blue-600 text-white shadow' : 'bg-white border border-slate-200 text-slate-500 hover:border-blue-300 hover:bg-blue-50'}`}>
                                                {f.label}
                                            </button>
                                        ))}
                                    </div>
                                    <button onClick={loadAllData} className="flex items-center gap-2 text-xs font-bold text-slate-500 hover:text-blue-600 bg-white px-3 py-2 rounded-lg border border-slate-200 shadow-sm whitespace-nowrap"><Icons.Refresh /> 最新に更新</button>
                                </div>
                                <div className="overflow-x-auto">
                                    <table className="w-full text-left">
                                        <thead>

                                                                    className="w-5 h-5 text-blue-600 focus:ring-blue-500 rounded border-slate-300"
                                                                />
                                                                <div className="flex-1 overflow-hidden">
                                                                    <div className="text-[10px] font-mono font-black text-slate-400 tracking-wider mb-0.5">{p.no}</div>
                                                                    <div className="text-sm font-bold text-slate-700 truncate">{p.name}</div>
                                                                </div>
                                                            </label>
                                                        ))}
                                                    </div>
                                                </div>
                                            )}
                                        </div>
                                    </div>
                                    {/* 明細情報 */}
                                    <div className="space-y-6">
                                        <div className="flex items-center gap-2 text-orange-600 font-black text-lg border-b-2 border-orange-100 pb-2 mb-4">
                                            <Icons.Plus /> 今回の請求・明細
                                        </div>
                                        <div className="grid grid-cols-2 gap-5">
                                            <div><label className="text-xs font-bold text-slate-400 mb-1 block uppercase tracking-widest">請求日 *</label>
                                                <input type="date" value={formData.billDate} onChange={e => setFormData({...formData, billDate:e.target.value})} className="w-full border-2 border-slate-200 rounded-xl p-3 text-sm font-bold focus:border-blue-500 outline-none shadow-sm" />
                                            </div>
                                            <div><label className="text-xs font-bold text-slate-400 mb-1 block uppercase tracking-widest">入金予定日</label>
                                                <input type="date" value={formData.payDate} onChange={e => setFormData({...formData, payDate:e.target.value})} className="w-full border-2 border-slate-200 rounded-xl p-3 text-sm font-bold focus:border-blue-500 outline-none shadow-sm" />
                                            </div>
                                        </div>
                                        <div className="grid grid-cols-2 gap-5">
                                            <div>
                                                <label className="text-xs font-bold text-slate-400 mb-1 block uppercase tracking-widest">
                                                    対象月 {formData.type === '出来高' && <span className="text-orange-600">*</span>}
                                                </label>
                                                <input 
                                                    type="month" 
                                                    value={formData.month} 
                                                    onChange={e => setFormData({...formData, month:e.target.value})} 
                                                    className="w-full border-2 border-slate-200 rounded-xl p-3 text-base font-bold outline-none shadow-sm focus:border-blue-500 h-[52px]" 
                                                />
                                            </div>
                                            <div><label className="text-xs font-black text-orange-600 mb-1 block uppercase tracking-widest">今回請求額 (税込) *</label>
                                                <input type="number" value={formData.amount} onChange={e => setFormData({...formData, amount:e.target.value})} className="w-full border-2 border-orange-300 rounded-xl p-3 text-xl font-black focus:border-orange-500 outline-none bg-orange-50 text-orange-700 font-mono shadow-sm h-[52px]" />
                                            </div>
                                        </div>
                                        <div><label className="text-xs font-bold text-slate-400 mb-1 block uppercase tracking-widest">事務用メモ (備考)</label>
                                            <textarea rows="3" value={formData.memo} onChange={e => setFormData({...formData, memo:e.target.value})} className="w-full border-2 border-slate-200 rounded-xl p-3 text-sm font-bold focus:border-blue-500 outline-none shadow-sm" placeholder="社内での管理用メモとして自由に入力してください"></textarea>
                                        </div>
                                    </div>
                                </div>
                                <div className="p-6 border-t-2 border-slate-100 bg-slate-50 flex justify-end gap-4">
                                    <button onClick={() => setIsModalOpen(false)} className="px-8 py-3 rounded-xl font-bold text-base text-slate-500 hover:bg-white border border-transparent hover:border-slate-200 transition-all">キャンセル</button>
                                    <button onClick={handleSave} disabled={isSaving} className="px-10 py-3 bg-blue-600 text-white font-black text-base rounded-xl shadow-lg hover:bg-blue-700 hover:-translate-y-0.5 active:translate-y-0 disabled:opacity-50 transition-all flex items-center gap-2">
                                        {isSaving ? <Icons.Loader /> : <Icons.Check />} データを登録・保存
                                    </button>
                                </div>
                            </div>
                        </div>
                    )}
                </div>
            );
        };

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<ErrorBoundary><App /></ErrorBoundary>);
    </script>
